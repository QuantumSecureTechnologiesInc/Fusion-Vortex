5. Verification: The "Hello Web" TestCreate a temporary test file to verify the HTTP server works with JSON.File: crates/http/examples/simple_server.rsuse fusion_http::{Server, Response};
use fusion_json;
use fusion_core::types::hybrid::HybridValue;

#[tokio::main]
async fn main() {
    let server = Server::new(|req| {
        println!("Received request: {} {}", req.method, req.path);
        
        // Create a Fusion Value
        let response_data = HybridValue::String("Hello from Fusion Web!".into());
        
        // Serialize to JSON
        let json_body = fusion_json::to_string(&response_data).unwrap();
        
        Response::ok()
            .body(json_body)
    });

    server.listen("127.0.0.1:8080").await.unwrap();
}
Run it:cargo run -p fusion_http --example simple_server
Then query it:curl http://localhost:8080
# Output: "Hello from Fusion Web!"
