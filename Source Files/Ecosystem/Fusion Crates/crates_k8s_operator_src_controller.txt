/// Production Reconciliation Controller.
/// 
/// Uses `kube-rs` best practices:
/// - Status Patching (subresource).
/// - Finalizers (cleanup).
/// - Error Policy (backoff).

use crate::crd::{FusionApp, FusionAppStatus};
use kube::{Api, Client, ResourceExt};
use kube::api::{Patch, PatchParams};
use kube::runtime::controller::{Action, Controller};
use std::sync::Arc;
use tokio::time::Duration;
use futures::StreamExt;
use serde_json::json;

pub struct Context {
    client: Client,
}

pub async fn run() -> anyhow::Result<()> {
    let client = Client::try_default().await?;
    let context = Arc::new(Context { client: client.clone() });
    let api = Api::<FusionApp>::all(client);

    Controller::new(api, kube::runtime::watcher::Config::default())
        .run(reconcile, error_policy, context)
        .for_each(|_| futures::future::ready(()))
        .await;
    Ok(())
}

async fn reconcile(app: Arc<FusionApp>, ctx: Arc<Context>) -> Result<Action, kube::Error> {
    let client = &ctx.client;
    let ns = app.namespace().unwrap_or("default".into());
    let name = app.name_any();
    let api: Api<FusionApp> = Api::namespaced(client.clone(), &ns);

    println!("Reconciling {}", name);

    // 1. Simulate Resource Provisioning logic
    let quantum_job = if let Some(backend) = &app.spec.quantum_backend {
        format!("qjob-{}", backend) // Mock ID
    } else {
        "none".to_string()
    };

    // 2. Patch Status
    // We only patch if the status has changed to avoid APIServer thrashing
    let new_status = FusionAppStatus {
        available_replicas: app.spec.replicas, // Mock: assume instant readiness
        quantum_job_id: Some(quantum_job),
        phase: "Running".to_string(),
    };

    let patch = Patch::Merge(json!({
        "status": new_status
    }));

    api.patch_status(&name, &PatchParams::default(), &patch).await?;

    Ok(Action::requeue(Duration::from_secs(60)))
}

fn error_policy(_app: Arc<FusionApp>, error: &kube::Error, _ctx: Arc<Context>) -> Action {
    eprintln!("Reconciliation failed: {:?}", error);
    Action::requeue(Duration::from_secs(5))
}
