// Polish: Uses real `toml` deserialization.

use std::fs;
use std::path::Path;
use serde::Deserialize;
use std::collections::HashMap;

#[derive(Deserialize, Debug)]
pub struct Manifest {
    pub package: Package,
    pub dependencies: Option<HashMap<String, String>>,
}

#[derive(Deserialize, Debug)]
pub struct Package {
    pub name: String,
    pub version: String,
    pub target: Option<String>,
}

impl Manifest {
    pub fn parse<P: AsRef<Path>>(path: P) -> Result<Self, String> {
        let p = path.as_ref();
        if !p.exists() {
            return Err(format!("Manifest not found at {:?}", p));
        }
        let content = fs::read_to_string(p).map_err(|e| e.to_string())?;
        toml::from_str(&content).map_err(|e| format!("Syntax Error: {}", e))
    }
}
