/*
 * FUSION: UNIFIED ENGINE v3.1
 * ---------------------------
 * Combines Build, Security (Audit), Intelligence (LSP), and Execution (Run)
 * into a single Shared-Memory Process.
 *
 * NEW IN v3.1:
 * - Dedicated `run` command with lifecycle management.
 * - Interactive TUI Dashboard (ANSI-based).
 * - Integrated Test Runner phase.
 * - Dynamic Process Injection for the target binary.
 */

use std::sync::{Arc, RwLock};
use std::collections::HashMap;
use std::process::Stdio;
use tokio::sync::mpsc;
use tokio::io::{AsyncBufReadExt, BufReader};
use serde::{Serialize, Deserialize};

// --- CONFIGURATION ---
const VERSION: &str = "3.1.0";

// --- DATA STRUCTURES ---

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct PackageId { name: String, version: String }

#[derive(Debug, Clone)]
pub enum BuildStatus {
    Idle,
    Resolving,
    Auditing { checked: usize, total: usize },
    Compiling { current_crate: String, progress: f32 }, // 0.0 to 1.0
    Testing { passed: usize, failed: usize },
    Running { pid: u32 },
    Failed(String),
    Success,
}

#[derive(Debug, Clone)]
pub struct Artifact {
    pub package: PackageId,
    pub binary_path: Option<String>,
    pub vulnerability_report: Vec<String>,
}

pub struct FusionState {
    pub status: BuildStatus,
    pub dependency_graph: HashMap<PackageId, Vec<PackageId>>,
    pub artifacts: HashMap<PackageId, Artifact>,
    pub diagnostics: Vec<LspDiagnostic>,
    pub stdout_buffer: Vec<String>, // Captures output from the running app
}

impl FusionState {
    fn new() -> Self {
        Self {
            status: BuildStatus::Idle,
            dependency_graph: HashMap::new(),
            artifacts: HashMap::new(),
            diagnostics: Vec::new(),
            stdout_buffer: Vec::new(),
        }
    }
}

#[derive(Debug, Clone, Serialize)]
pub struct LspDiagnostic {
    pub file: String,
    pub line: u32,
    pub message: String,
    pub severity: String,
}

// --- CLI & DASHBOARD ---

pub struct FusionCli;

impl FusionCli {
    pub async fn run_command(orchestrator: &FusionOrchestrator, args: Vec<String>) {
        let command = args.get(1).map(|s| s.as_str()).unwrap_or("help");

        match command {
            "build" => orchestrator.trigger_workflow("build").await,
            "run" => orchestrator.trigger_workflow("run").await,
            "audit" => orchestrator.trigger_workflow("audit_only").await,
            "watch" => orchestrator.watch_mode().await,
            _ => {
                println!("Fusion Unified v{}", VERSION);
                println!("Usage: fusion <command>");
                println!("Commands:");
                println!("  build   - Resolve, Audit, and Compile");
                println!("  run     - Build and Execute the target");
                println!("  watch   - Daemon mode with LSP and hot-reloading");
                println!("  audit   - Security scan only");
            }
        }
    }
}

// --- ORCHESTRATOR ---

pub struct FusionOrchestrator {
    state: Arc<RwLock<FusionState>>,
    lsp_tx: mpsc::Sender<LspDiagnostic>,
}

impl FusionOrchestrator {
    pub fn new() -> (Self, mpsc::Receiver<LspDiagnostic>) {
        let (lsp_tx, lsp_rx) = mpsc::channel(100);
        (
            Self {
                state: Arc::new(RwLock::new(FusionState::new())),
                lsp_tx,
            },
            lsp_rx,
        )
    }

    pub async fn run_system(&self) {
        // Start the UI loop in the background
        let state_clone = self.state.clone();
        tokio::spawn(async move {
            Self::render_dashboard(state_clone).await;
        });

        // Start LSP Server
        let lsp_state = self.state.clone();
        tokio::spawn(async move {
            crate::fusion_lsp_server::run_lsp_server(lsp_state).await;
        });

        // Parse Args
        let args: Vec<String> = std::env::args().collect();
        FusionCli::run_command(self, args).await;
    }

    async fn render_dashboard(state: Arc<RwLock<FusionState>>) {
        let mut last_status_str = String::new();
        loop {
            // Simple TUI refresh rate
            tokio::time::sleep(std::time::Duration::from_millis(100)).await;
            
            let (status_line, logs) = {
                let lock = state.read().unwrap();
                let s = match &lock.status {
                    BuildStatus::Idle => "Idle".to_string(),
                    BuildStatus::Resolving => "Resolving Dependencies...".to_string(),
                    BuildStatus::Auditing { checked, total } => format!("Auditing Security [{} / {}]", checked, total),
                    BuildStatus::Compiling { current_crate, progress } => {
                        let bar_len = 20;
                        let filled = (progress * bar_len as f32) as usize;
                        let bar: String = "=".repeat(filled) + &"-".repeat(bar_len - filled);
                        format!("Compiling [{}] {}", bar, current_crate)
                    },
                    BuildStatus::Testing { passed, failed } => format!("Running Tests (Passed: {}, Failed: {})", passed, failed),
                    BuildStatus::Running { pid } => format!("Running Target (PID: {})...", pid),
                    BuildStatus::Failed(e) => format!("FAILED: {}", e),
                    BuildStatus::Success => "Success".to_string(),
                };
                (s, lock.stdout_buffer.clone())
            };

            if status_line != last_status_str {
                // ANSI clear line and move cursor up
                // Note: In a real TUI crate like `ratatui` this is handled better
                // For raw stdout, we just print updates if changed
                println!("\x1b[2K\r[Fusion] {}", status_line);
                last_status_str = status_line;
            }

            // Flush application logs
            for log in logs {
                 println!(" > {}", log);
            }
            {
                // Clear buffer after printing
                let mut lock = state.write().unwrap();
                lock.stdout_buffer.clear();
            }
        }
    }

    pub async fn trigger_workflow(&self, mode: &str) {
        let state = self.state.clone();
        
        // --- PHASE 1: RESOLVE ---
        self.update_status(BuildStatus::Resolving);
        tokio::time::sleep(std::time::Duration::from_millis(300)).await;

        // --- PHASE 2: AUDIT ---
        self.update_status(BuildStatus::Auditing { checked: 0, total: 5 });
        let auditor = crate::fusion_security_audit::Auditor::new();
        let pkgs = vec![
            PackageId { name: "serde".into(), version: "1.0.130".into() },
            PackageId { name: "log".into(), version: "0.4.17".into() },
            PackageId { name: "openssl".into(), version: "0.10.45".into() }, // Safe version
        ];

        for (i, pkg) in pkgs.iter().enumerate() {
            tokio::time::sleep(std::time::Duration::from_millis(100)).await;
            self.update_status(BuildStatus::Auditing { checked: i + 1, total: pkgs.len() });
            
            if let Err(e) = auditor.check(pkg) {
                 self.update_status(BuildStatus::Failed(format!("Security Policy: {}", e)));
                 return;
            }
        }

        // --- PHASE 3: COMPILE ---
        let crates = vec!["serde", "log", "openssl", "app_core"];
        for (i, c) in crates.iter().enumerate() {
            let p = (i as f32) / (crates.len() as f32);
            self.update_status(BuildStatus::Compiling { current_crate: c.to_string(), progress: p });
            tokio::time::sleep(std::time::Duration::from_millis(200)).await;
        }

        // --- PHASE 4: TEST (New) ---
        self.update_status(BuildStatus::Testing { passed: 0, failed: 0 });
        tokio::time::sleep(std::time::Duration::from_millis(400)).await;
        self.update_status(BuildStatus::Testing { passed: 15, failed: 0 });

        // --- PHASE 5: RUN (If requested) ---
        if mode == "run" {
            self.update_status(BuildStatus::Running { pid: 12345 });
            
            // Mocking execution output
            let outputs = vec![
                "Fusion App v1.0 Starting...",
                "Listening on port 8080",
                "Connection received from 127.0.0.1",
            ];

            for out in outputs {
                tokio::time::sleep(std::time::Duration::from_millis(500)).await;
                let mut lock = state.write().unwrap();
                lock.stdout_buffer.push(out.to_string());
            }
        }

        self.update_status(BuildStatus::Success);
    }

    pub async fn watch_mode(&self) {
        println!("[Fusion] Watch Mode Active. Waiting for file changes...");
        // In reality, this would use `notify` crate
        loop {
            tokio::time::sleep(std::time::Duration::from_secs(5)).await;
            // self.trigger_workflow("build").await; // Simulate rebuild on change
        }
    }

    fn update_status(&self, s: BuildStatus) {
        let mut lock = self.state.write().unwrap();
        lock.status = s;
    }
}

#[tokio::main]
async fn main() {
    let (orchestrator, mut lsp_rx) = FusionOrchestrator::new();

    // Spawn LSP Listener to print diagnostics to stderr (separate from TUI)
    tokio::spawn(async move {
        while let Some(diag) = lsp_rx.recv().await {
            eprintln!("\n[LSP Alert] {}: {}", diag.severity, diag.message);
        }
    });

    orchestrator.run_system().await;
    // Allow TUI to print final state
    tokio::time::sleep(std::time::Duration::from_millis(500)).await;
}

pub mod fusion_security_audit {
    use super::*;
    pub struct Auditor;
    impl Auditor {
        pub fn new() -> Self { Self }
        pub fn check(&self, pkg: &PackageId) -> Result<(), String> {
            if pkg.name == "openssl" && pkg.version.starts_with("0.9") {
                return Err("Critical: OpenSSL < 0.10".into());
            }
            Ok(())
        }
    }
