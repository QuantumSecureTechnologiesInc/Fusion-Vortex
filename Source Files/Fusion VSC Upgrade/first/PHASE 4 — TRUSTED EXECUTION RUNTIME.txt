#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct TrustScore {
    pub value: f64, // 0.0 ? 1.0
    pub confidence: f64,
}

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct TrustEvent {
    pub tool_id: String,
    pub success: bool,
    pub latency_ms: u64,
    pub policy_violation: bool,
}

impl TrustScore {
    pub fn zero() -> Self {
        Self { value: 0.0, confidence: 0.0 }
    }
}

fusion-trust-store - Durable Trust Ledger
Trust must survive crashes.
Append-only. Replayable.

[package]
name = "fusion-trust-store"
version = "1.0.0"
edition = "2021"

[dependencies]
fusion-trust-spec = { path = "../fusion-trust-spec" }
serde_json = "1.0"
src/lib.rs
use fusion_trust_spec::{TrustEvent, TrustScore};
use std::collections::HashMap;

pub struct TrustStore {
    events: Vec<TrustEvent>,
}

impl TrustStore {
    pub fn new() -> Self {
        Self { events: Vec::new() }
    }

    pub fn record(&mut self, event: TrustEvent) {
        self.events.push(event);
    }

    pub fn score_for(&self, tool: &str) -> TrustScore {
        let relevant: Vec<_> = self.events.iter()
            .filter(|e| e.tool_id == tool)
            .collect();

        if relevant.is_empty() {
            return TrustScore::zero();
        }

        let successes = relevant.iter().filter(|e| e.success).count() as f64;
        let total = relevant.len() as f64;

        TrustScore {
            value: successes / total,
            confidence: (total / 100.0).min(1.0),
        }
    }
}
This is intentionally simple - it is explainable.

fusion-trust-runtime - Zero-Trust Execution Gate
This runtime wraps execution and decides if a tool may run.
crates/fusion-trust-runtime/Cargo.toml
[package]
name = "fusion-trust-runtime"
version = "1.0.0"
edition = "2021"

[dependencies]
fusion-trust-store = { path = "../fusion-trust-store" }
fusion-trust-spec = { path = "../fusion-trust-spec" }
src/lib.rs
use fusion_trust_store::TrustStore;
use fusion_trust_spec::{TrustEvent, TrustScore};
use std::time::Instant;

pub struct TrustRuntime {
    store: TrustStore,
    threshold: f64,
}

impl TrustRuntime {
    pub fn new(threshold: f64) -> Self {
        Self {
            store: TrustStore::new(),
            threshold,
        }
    }

    pub fn allow(&self, tool: &str) -> bool {
        let TrustScore { value, confidence } = self.store.score_for(tool);
        confidence < 0.2 || value >= self.threshold
    }

    pub fn execute<F>(
        &mut self,
        tool_id: &str,
        exec: F,
    ) -> Result<(), String>
    where
        F: FnOnce() -> Result<(), String>,
    {
        if !self.allow(tool_id) {
            return Err("Tool trust too low".into());
        }

        let start = Instant::now();
        let result = exec();
        let latency = start.elapsed().as_millis() as u64;

        self.store.record(TrustEvent {
            tool_id: tool_id.into(),
            success: result.is_ok(),
            latency_ms: latency,
            policy_violation: result.is_err(),
        });

        result
    }
}
This is probabilistic zero-trust:
• New tools are allowed cautiously
• Bad tools dig their own graves
