/// Helper to draw the header with status buttons
fn draw_header<B: ratatui::backend::Backend>(f: &mut ratatui::Frame<B>, state: &AppState, area: Rect) {
    let header_layout = Layout::default()
        .direction(Direction::Horizontal)
        .constraints([
            Constraint::Percentage(30), // Logo and Welcome
            Constraint::Percentage(70), // Status Buttons
        ])
        .split(area);

    // Left side: Logo and Welcome
    let logo_welcome = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Length(2), // Logo
            Constraint::Length(3), // Welcome
        ])
        .split(header_layout[0]);

    // Fusion Logo
    let logo_text = Paragraph::new(FUSION_LOGO_TEXT[0])
        .style(Style::default().fg(Color::Cyan).add_modifier(Modifier::BOLD))
        .alignment(Alignment::Left);
    f.render_widget(logo_text, logo_welcome[0]);

    // Welcome Text
    let welcome_text = Paragraph::new(vec![
        Line::from(Span::styled(format!("Welcome back, {}!", state.user_name), Style::default().fg(Color::White))),
        Line::from(Span::styled("Mode: Planning", Style::default().fg(Color::Gray))),
        Line::from(Span::styled("Mode. Dirorining", Style::default().fg(Color::Gray))),
    ])
    .alignment(Alignment::Left);
    f.render_widget(welcome_text, logo_welcome[1]);

    // Right side: Status Buttons
    let buttons_layout = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Length(2), // Top row buttons
            Constraint::Length(2), // Bottom row buttons
        ])
        .split(header_layout[1]);

    let top_buttons = Layout::default()
        .direction(Direction::Horizontal)
        .constraints([
            Constraint::Percentage(50),
            Constraint::Percentage(50),
        ])
        .split(buttons_layout[0]);

    let bottom_buttons = Layout::default()
        .direction(Direction::Horizontal)
        .constraints([
            Constraint::Percentage(50),
            Constraint::Percentage(50),
        ])
        .split(buttons_layout[1]);

    // Draw status buttons
    for (i, (text, _)) in STATUS_BUTTONS.iter().enumerate() {
        let button_area = if i == 0 { top_buttons[0] } 
                        else if i == 1 { top_buttons[1] }
                        else if i == 2 { bottom_buttons[0] }
                        else { bottom_buttons[1] };
        
        let is_active = i < 2; // First two buttons are active
        let button_style = if is_active {
            Style::default().fg(Color::Cyan).add_modifier(Modifier::BOLD)
        } else {
            Style::default().fg(Color::Gray)
        };
        
        let button = Paragraph::new(vec![Line::from(Span::styled(*text, button_style))])
            .block(Block::default().borders(Borders::ALL).border_style(button_style));
        f.render_widget(button, button_area);
    }

    // Header text
    let header_text = Paragraph::new(vec![
        Line::from(Span::styled(&state.header_text, Style::default().fg(Color::Gray))),
    ])
    .alignment(Alignment::Center);
    f.render_widget(header_text, area);
}

// Shared Input Bar Logic (Bottom of screen)
fn draw_input_bar<B: ratatui::backend::Backend>(f: &mut ratatui::Frame<B>, state: &AppState, area: Rect) {
    // 1. Draw Command Palette Overlay if active
    if state.command_palette_active {
        let palette_height = SLASH_COMMANDS.len() as u16 + 2;
        let palette_area = Rect::new(
            area.x + 2,  
            area.y.saturating_sub(palette_height),  
            30, // Fixed width for palette
            palette_height, 
        );

        let palette_items: Vec<ListItem> = SLASH_COMMANDS.iter().enumerate().map(|(i, (cmd, _))| {
            let style = if state.command_palette_index == Some(i) {
                Style::default().bg(Color::Cyan).fg(Color::Black)
            } else {
                Style::default().fg(Color::White)
            };
            ListItem::new(Span::styled(*cmd, style))
        }).collect();

        let palette_block = Block::default()
            .borders(Borders::ALL)
            .border_type(BorderType::Rounded)
            .style(Style::default().bg(Color::Black));
        
        f.render_widget(Clear, palette_area);
        f.render_widget(List::new(palette_items).block(palette_block), palette_area);
    }

    // 2. Draw Task Group Dropdown if active
    if state.task_group_dropdown_active {
        let dropdown_height = 4;
        let dropdown_area = Rect::new(
            area.x + 2,
            area.y.saturating_sub(dropdown_height),
            15,
            dropdown_height,
        );
        draw_task_group_dropdown(f, state, dropdown_area);
    }

    // 3. Draw Project List if active
    if state.project_list_active {
        let project_height = 6;
        let project_area = Rect::new(
            area.x + 2,
            area.y.saturating_sub(project_height),
            40,
            project_height,
        );
        draw_project_list(f, state, project_area);
    }

    // 4. Draw Input Line
    let prompt = Span::styled("> ", Style::default().fg(Color::Gray));
    let input_text = Span::styled(&state.input, Style::default().fg(Color::White));
    let cursor = Span::styled("█", Style::default().fg(Color::Cyan));

    let input_line = Line::from(vec![prompt, input_text, cursor]);
    let input_widget = Paragraph::new(input_line);
    f.render_widget(input_widget, area);
}

// --- MAIN LOOP ---

fn run_app<B: ratatui::backend::Backend>(terminal: &mut Terminal<B>, state: &mut AppState) -> io::Result<()> {
    loop {
        terminal.draw(|f| match state.current_view {
            View::Dashboard => draw_dashboard(f, state),
            View::AgentReview => draw_agent_review(f, state),
            View::TaskGroup => draw_task_group(f, state),
            View::Welcome => draw_welcome(f, state),
            View::Exit => {},
        })?;

        if let Event::Key(key) = event::read()? {
            match key.code {
                KeyCode::Enter => {
                    // Simple navigation simulation based on input
                    match state.input.trim() {
                        "/ask" | "ask" => {
                            state.current_view = View::TaskGroup;
                            state.header_text = "TASK GROUP MODE".to_string();
                        },
                        "/review" | "review" => {
                            state.current_view = View::AgentReview;
                            state.header_text = "AGENT REVIEW".to_string();
                        },
                        "home" | "/home" => {
                            state.current_view = View::Dashboard;
                            state.header_text = "AGENT REVIEW CLI".to_string();
                        },
                        "welcome" | "/welcome" => {
                            state.current_view = View::Welcome;
                            state.header_text = "".to_string();
                        },
                        "exit" | "/exit" => return Ok(()),
                        _ => {
                            // If palette was active, select item
                            if state.command_palette_active {
                                if let Some(idx) = state.command_palette_index {
                                    state.input = SLASH_COMMANDS[idx].0.to_string();
                                    state.command_palette_active = false;
                                }
                            } else {
                                state.input.clear();
                            }
                        }
                    }
                }
                KeyCode::Esc => {
                    if state.command_palette_active {
                        state.command_palette_active = false;
                    } else if state.task_group_dropdown_active {
                        state.task_group_dropdown_active = false;
                    } else if state.project_list_active {
                        state.project_list_active = false;
                    } else if !state.input.is_empty() {
                        state.input.clear();
                    } else {
                        // Fallback to dashboard if ESC pressed on other screens
                        if state.current_view != View::Dashboard {
                            state.current_view = View::Dashboard;
                            state.header_text = "AGENT REVIEW CLI".to_string();
                        }
                    }
                }
                KeyCode::Char(c) => {
                    state.input.push(c);
                    if c == '/' {
                        state.command_palette_active = true;
                        state.command_palette_index = Some(0);
                    } else if c == 't' && state.input.ends_with("task") {
                        state.task_group_dropdown_active = true;
                        state.task_group_index = Some(0);
                    } else if c == 'p' && state.input.ends_with("project") {
                        state.project_list_active = true;
                        state.project_index = Some(0);
                    } else if !state.input.starts_with('/') {
                        state.command_palette_active = false;
                    }
                }
                KeyCode::Backspace => {
                    state.input.pop();
                    if state.input.is_empty() {
                        state.command_palette_active = false;
                        state.task_group_dropdown_active = false;
                        state.project_list_active = false;
                    }
                }
                KeyCode::Down => {
                    if state.command_palette_active {
                        let i = state.command_palette_index.unwrap_or(0);
                        state.command_palette_index = Some((i + 1) % SLASH_COMMANDS.len());
                    } else if state.task_group_dropdown_active {
                        let i = state.task_group_index.unwrap_or(0);
                        state.task_group_index = Some((i + 1) % 3);
                    } else if state.project_list_active {
                        let i = state.project_index.unwrap_or(0);
                        state.project_index = Some((i + 1) % 4);
                    }
                }
                KeyCode::Up => {
                    if state.command_palette_active {
                        let i = state.command_palette_index.unwrap_or(0);
                        state.command_palette_index = Some(if i == 0 { SLASH_COMMANDS.len() - 1 } else { i - 1 });
                    } else if state.task_group_dropdown_active {
                        let i = state.task_group_index.unwrap_or(0);
                        state.task_group_index = Some(if i == 0 { 2 } else { i - 1 });
                    } else if state.project_list_active {
                        let i = state.project_index.unwrap_or(0);
                        state.project_index = Some(if i == 0 { 3 } else { i - 1 });
                    }
                }
                _ => {}
            }
        }
        
        if state.current_view == View::Exit {
            return Ok(());
        }
    }
}

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let _ = Cli::parse(); // Required for compatibility

    enable_raw_mode()?;
    let mut stdout = io::stdout();
    execute!(stdout, EnterAlternateScreen, EnableMouseCapture)?;
    let backend = CrosstermBackend::new(stdout);
    let mut terminal = Terminal::new(backend)?;
    let mut state = AppState::default();
    let res = run_app(&mut terminal, &mut state);

    disable_raw_mode()?;
    execute!(terminal.backend_mut(), LeaveAlternateScreen, DisableMouseCapture)?;
    terminal.show_cursor()?;

    if let Err(err) = res {
        println!("{:?}", err);
    }

    Ok(())
}

#[derive(Parser, Debug)]
#[command(author, version, about = "Fusion VSC CLI", long_about = None)]
struct Cli {
    #[command(subcommand)]
    command: Option<Commands>,
}

enum Commands {
    Menu
