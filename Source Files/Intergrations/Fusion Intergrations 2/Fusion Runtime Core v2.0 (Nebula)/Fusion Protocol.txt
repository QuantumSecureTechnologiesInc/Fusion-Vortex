syntax = "proto3";
package fusion_core_v2;

service Runtime {
  // Standard liveness check
  rpc HealthCheck (HealthCheckRequest) returns (HealthCheckResponse);
  
  // upload and execute a plugin in one go (stateless execution)
  rpc ExecutePlugin (PluginRequest) returns (PluginResponse);
}

message HealthCheckRequest {
  string client_id = 1;
}

message HealthCheckResponse {
  string status = 1;
  string version = 2;
  double load_index = 3;
}

message PluginRequest {
  string plugin_name = 1;
  // The raw WASM binary bytes
  bytes wasm_binary = 2; 
  // Input data to pass to the plugin
  string input_data = 3;
}

message PluginResponse {
  int32 exit_code = 1;
  string output_data = 2;
  string error_message = 3;
  double execution_time_ms = 4;
}
