/// Attention Mask Generator.
/// 
/// Generates masks for Causal, Padding, and Sliding Window Attention (SWA).

use fusion_core::types::tensor::Matrix;
use fusion_core::FusionResult;
use std::f64::NEG_INFINITY;

pub struct AttentionMaskGenerator;

impl AttentionMaskGenerator {
    /// Generate a Sliding Window Attention mask.
    /// Limits attention to a fixed window (W) around each token (Mistral/Longformer).
    /// Returns mask where 0.0 means allow, NEG_INFINITY means mask out.
    pub fn generate_sliding_window_mask(seq_len: usize, window_size: usize) -> FusionResult<Matrix<f64>> {
        let mut mask = Matrix::new(vec![0.0; seq_len * seq_len], [seq_len, seq_len])?;
        
        for r in 0..seq_len {
            for c in 0..seq_len {
                // Check 1: Causal mask (c > r means lookahead is masked)
                let is_causal = c > r;
                
                // Check 2: Sliding Window (c < r - window_size means mask)
                let is_outside_window = (r as i32 - c as i32) > window_size as i32;
                
                if is_causal || is_outside_window {
                    mask.set([r, c], NEG_INFINITY)?; 
                } else {
                    mask.set([r, c], 0.0)?;
                }
            }
        }
        Ok(mask)
    }
}
