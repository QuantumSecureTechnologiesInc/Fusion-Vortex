/// Mapping Fusion Core types to C-compatible equivalents.

use libc::{c_char, c_int, c_double, c_void};

/// Represents a Fusion Integer in C (64-bit signed).
pub type CInt = i64;

/// Represents a Fusion Float in C (64-bit double).
pub type CFloat = c_double;

/// Represents a Fusion Boolean in C (int, 0=false, 1=true).
pub type CBool = c_int;

/// Opaque pointer to a Fusion Struct (heap allocated).
pub type CStructPtr = *mut c_void;

/// Raw C-String pointer.
pub type CStringPtr = *const c_char;
