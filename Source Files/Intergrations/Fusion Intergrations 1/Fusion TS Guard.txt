/**
 * Fusion TS Guard
 * ---------------
 * Validation logic hardened for the Hybrid Crypto environment.
 * Includes specific checks for "50/50" key structures.
 */

export class Guard {
  
  /**
   * Ensures a value is not null or undefined.
   */
  static required<T>(value: T | null | undefined, name: string): T {
    if (value === null || value === undefined) {
      throw new Error(`Guard Validation Failed: '${name}' is required.`);
    }
    return value;
  }

  /**
   * Validates that a key bundle adheres to the Fusion 50/50 Standard.
   * Must contain both 'classical' (ECC) and 'quantum' (Lattice) components.
   */
  static isHybridKeyBundle(bundle: any): boolean {
    if (!bundle || typeof bundle !== 'object') return false;
    
    // Check Classical
    const hasClassical = bundle.classical && 
      (bundle.classical.kty === 'EC' || bundle.classical instanceof CryptoKey);
    
    // Check Quantum (MiniKyber shape: { matrixA, vectorT } or { u, v })
    // We check for generic structure properties
    const hasQuantum = bundle.quantum && typeof bundle.quantum === 'object';

    return !!(hasClassical && hasQuantum);
  }

  /**
   * Enforces specific byte length for buffers (e.g. SHA-256 output).
   */
  static byteLength(buffer: Uint8Array, expected: number, name: string): void {
    if (buffer.byteLength !== expected) {
      throw new Error(`Guard: '${name}' must be ${expected} bytes, got ${buffer.byteLength}`);
    }
  }
}
