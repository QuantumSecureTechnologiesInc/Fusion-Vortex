// Status: Runtime Core
// Purpose: Execution context for Flux-Resolve with Backpressure.
// Polish: Added specific Error enums and thread-safe locking logic.

use std::collections::VecDeque;
use std::sync::{Arc, Mutex};
use crate::flux::resolve::GradientMerger;

#[derive(Debug)]
pub enum EngineError {
    Backpressure(String),
    QueueFull,
}

/// The runtime engine that drives the resolution process.
pub struct ResolutionEngine {
    queue: Arc<Mutex<VecDeque<Vec<f32>>>>,
    merger: GradientMerger,
    max_queue_bytes: usize,
    current_bytes: Arc<Mutex<usize>>,
}

impl ResolutionEngine {
    pub fn new(max_memory_mb: usize) -> Self {
        Self {
            queue: Arc::new(Mutex::new(VecDeque::new())),
            merger: GradientMerger::new(),
            max_queue_bytes: max_memory_mb * 1024 * 1024,
            current_bytes: Arc::new(Mutex::new(0)),
        }
    }

    /// Submits a gradient vector for resolution.
    pub fn submit(&self, gradient: Vec<f32>) -> Result<(), EngineError> {
        let size = gradient.len() * 4; // 4 bytes per f32
        let mut current = self.current_bytes.lock().map_err(|_| EngineError::QueueFull)?; // Lock poisoning handling

        if *current + size > self.max_queue_bytes {
            return Err(EngineError::Backpressure("Resolution queue exceeds memory limit".into()));
        }

        let mut q = self.queue.lock().map_err(|_| EngineError::QueueFull)?;
        q.push_back(gradient);
        *current += size;
        
        Ok(())
    }

    /// Processes the queue, merging gradients until empty.
    pub fn tick(&self, accumulator: &mut Vec<f32>) {
        // We use a block here to ensure locks are held for minimal time,
        // though `pop_front` requires holding it.
        let mut q = self.queue.lock().unwrap();
        let mut current = self.current_bytes.lock().unwrap();

        while let Some(grad) = q.pop_front() {
            let size = grad.len() * 4;
            
            if accumulator.len() != grad.len() {
                if accumulator.is_empty() {
                    *accumulator = grad;
                } else {
                    eprintln!("Gradient mismatch dropped: acc={} vs grad={}", accumulator.len(), size/4);
                }
            } else {
                self.merger.merge(accumulator, &grad);
            }

            *current -= size;
        }
    }
}
