//! src/macros.rs
//! Standard Macros for the Fusion Library.
//!
//! These macros allow developers to write "Code as Literature" more naturally,
//! removing the boilerplate of method calls for common narrative actions.

/// Records a simple verse in the current story.
///
/// # Example
/// ```ignore
/// verse!(story, "User logged in");
/// ```
#[macro_export]
macro_rules! verse {
    ($story:expr, $msg:expr) => {
        $story.record_verse($msg)
    };
    ($story:expr, $fmt:expr, $($arg:tt)*) => {
        $story.record_verse(&format!($fmt, $($arg)*))
    };
}

/// Records a verse with structural evidence (data).
///
/// # Example
/// ```ignore
/// verse_with!(story, "Transaction processed", transaction_data);
/// ```
#[macro_export]
macro_rules! verse_with {
    ($story:expr, $msg:expr, $data:expr) => {
        $story.record_verse_with_data($msg, $data)
    };
}

/// A Guard Clause macro.
/// Checks a condition; if false, concludes the story with failure and returns early.
///
/// Uses `ProcessingError` as the default failure type to avoid misleading "Access Denied" logs
/// for general logic failures.
///
/// # Example
/// ```ignore
/// ensure!(story, payload.is_valid(), "Payload checksum failed");
/// ```
#[macro_export]
macro_rules! ensure {
    ($story:expr, $condition:expr, $failure_reason:expr) => {
        if !$condition {
            // 1. Conclude the narrative first so the log is complete.
            let _ = $story.conclude_failure($failure_reason);
            
            // 2. Return the error.
            // We use ProcessingError as the safe default for general assertions.
            // The .into() allows this to convert into the function's Result type.
            return Err($crate::seal::SecurityViolation::ProcessingError(
                $failure_reason.to_string()
            ).into());
        }
    };
}
