// ion/src/lto.rs
// Cross-Language Linker Wrapper (lld) - Updated for Bitcode handling

use std::path::Path;
use std::process::Command;
use anyhow::{Result, bail, Context};
use colored::*;

pub fn link(inputs: &[&Path], output: &Path, lto_enabled: bool) -> Result<()> {
    println!("   {} Linking {} objects (LTO: {})...", "Holo-Link".magenta().bold(), inputs.len(), lto_enabled);

    let mut cmd = Command::new("clang++");
    
    for input in inputs {
        cmd.arg(input);
    }
    
    cmd.arg("-o").arg(output);
    cmd.arg("-fuse-ld=lld"); // Critical for performance
    
    if lto_enabled {
        // If inputs are Bitcode (.bc), Clang/LLD handles LTO automatically
        // but we force the flag to be explicit
        cmd.arg("-flto");
        cmd.arg("-O3"); // Link-time optimization level
    }
    
    let status = cmd.status().context("Failed to invoke linker")?;
    if !status.success() { bail!("Linking failed"); }
    Ok(())
}
