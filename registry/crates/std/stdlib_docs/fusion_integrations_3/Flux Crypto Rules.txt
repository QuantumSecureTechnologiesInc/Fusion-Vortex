/// E006: Enforce Hybrid 50/50 Crypto
pub fn check_crypto_compliance(ctx: &LinterContext, reporter: &mut DiagnosticReporter) {
    // 1. Check dependencies for legacy crypto libraries
    let legacy_crates = vec!["rsa", "openssl", "aes", "sha2"];
    let deps = ctx.manifest.dependencies.to_string(); // Simplified for demo
    
    for krate in legacy_crates {
        if deps.contains(krate) {
            reporter.add(Diagnostic {
                code: "E006".into(),
                level: Level::Warning,
                message: format!("Legacy crypto crate '{}' detected.", krate),
                file: Some("Fusion.toml".into()),
                line: None,
                hint: Some(format!("Fusion requires Hybrid 50/50 PQC. Use 'fusion_std::crypto' instead of '{}'.", krate)),
            });
        }
    }

    // 2. Scan source for usage of raw crypto primitives
    lazy_static! {
        static ref RSA_REGEX: Regex = Regex::new(r"(RsaPrivateKey|EVP_PKEY)").unwrap();
    }

    let src_dir = ctx.root.join("src");
    for entry in WalkDir::new(src_dir).into_iter().filter_map(|e| e.ok()) {
        if entry.path().extension().map_or(false, |ext| ext == "rs") {
            if let Ok(content) = fs::read_to_string(entry.path()) {
                for (i, line) in content.lines().enumerate() {
                    if RSA_REGEX.is_match(line) {
                        reporter.add(Diagnostic {
                            code: "E006".into(),
                            level: Level::Error,
                            message: "Non-Hybrid Crypto Primitive detected.".into(),
                            file: Some(entry.path().display().to_string()),
                            line: Some(i + 1),
                            hint: Some("All secrets must be protected by Quantum-Resistant layers via fusion_std::crypto.".into()),
                        });
                    }
                }
            }
        }
    }
}
