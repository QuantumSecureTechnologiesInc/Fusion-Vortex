// THE RESOLUTION: Real AEAD Encryption (No XOR)
// Dependencies: chacha20poly1305

use fusion_rt_core::crypto::{ProductionCrypto, CryptoDriver};
use fusion_rt_core::Result;
use chacha20poly1305::{ChaCha20Poly1305, Key, Nonce}; // Real AEAD
use chacha20poly1305::aead::{Aead, KeyInit};
use rand::{RngCore, rngs::OsRng};

pub struct Identity {
    // Stores full keypair logic
    pub(crate) inner: fusion_rt_core::crypto::HybridKeyPair,
}

impl Identity {
    pub fn generate() -> Result<Self> {
        let driver = ProductionCrypto::new();
        let keys = driver.generate_keypair()?;
        Ok(Self { inner: keys })
    }

    pub fn public_bytes(&self) -> Vec<u8> {
        let mut bytes = Vec::new();
        bytes.extend_from_slice(self.inner.classical_pk.as_bytes());
        bytes.extend_from_slice(self.inner.quantum_pk.as_bytes());
        bytes
    }
}

pub struct SecureChannel {
    cipher: ChaCha20Poly1305,
    // In production, manage nonces carefully (e.g., counter based)
    // For this stateless wrapper, we assume random nonces prepended to ciphertext
}

impl SecureChannel {
    pub fn connect(peer_identity_blob: &[u8]) -> Result<(Self, Vec<u8>)> {
        if peer_identity_blob.len() < 32 { return Err(fusion_rt_core::FusionError::ReactorError("Invalid Peer Key".into())); }
        
        let (c_part, q_part) = peer_identity_blob.split_at(32);

        let driver = ProductionCrypto::new();
        let (secret_wrapper, ciphertext) = driver.encapsulate(c_part, q_part)?;
        
        // Unsafe transmute is safe here because HybridSharedSecret is repr(transparent) [u8;32]
        let secret_bytes: [u8; 32] = unsafe { std::mem::transmute(secret_wrapper) };
        let key = Key::from_slice(&secret_bytes);
        let cipher = ChaCha20Poly1305::new(key);

        Ok((SecureChannel { cipher }, ciphertext))
    }

    /// Encrypts data using ChaCha20-Poly1305
    pub fn encrypt(&self, plaintext: &[u8]) -> Vec<u8> {
        let mut nonce_bytes = [0u8; 12];
        OsRng.fill_bytes(&mut nonce_bytes);
        let nonce = Nonce::from_slice(&nonce_bytes); // 96-bits; unique per message

        let ciphertext = self.cipher.encrypt(nonce, plaintext)
            .expect("Encryption failure"); // Should only fail on too large data

        // Prepend nonce to ciphertext
        let mut packet = Vec::with_capacity(12 + ciphertext.len());
        packet.extend_from_slice(&nonce_bytes);
        packet.extend_from_slice(&ciphertext);
        packet
    }

    /// Decrypts data using ChaCha20-Poly1305
    pub fn decrypt(&self, packet: &[u8]) -> Vec<u8> {
        if packet.len() < 12 { return vec![]; } // Error handling in prod would return Result

        let (nonce_bytes, ciphertext) = packet.split_at(12);
        let nonce = Nonce::from_slice(nonce_bytes);

        self.cipher.decrypt(nonce, ciphertext)
            .expect("Decryption failure / Auth Tag Mismatch")
    }
}
