pub(crate) fn rt() -> std::sync::Arc<fusion_rt_core::reactor::HyperRing> {
    fusion_rt_core::executor::get_reactor()
}

/// Internal helper to bridge Reactor Ops to Futures
pub(crate) async fn submit_op(op: RingOp) -> Result<RingResult> {
    let reactor = rt();
    
    // Create a generic waker hook
    // In a real system, the reactor needs a way to map the EventId back to this channel.
    // We simulate the mechanism of "register waker" -> "submit" -> "await".
    
    struct OpFuture {
        op: Option<RingOp>,
        id: Option<u64>,
    }
    
    impl std::future::Future for OpFuture {
        type Output = Result<RingResult>;
        fn poll(mut self: std::pin::Pin<&mut Self>, cx: &mut std::task::Context<'_>) -> std::task::Poll<Self::Output> {
            let reactor = rt();
            if let Some(op) = self.op.take() {
                let id = reactor.submit(op, cx.waker().clone());
                self.id = Some(id);
                std::task::Poll::Pending
            } else {
                // In a real reactor, we'd query the completion queue/map here
                // For the "Metal" demo, we assume the waker is triggered on completion.
                std::task::Poll::Ready(Ok(RingResult::Ok)) // Simplified flow
            }
        }
    }

    OpFuture { op: Some(op), id: None }.await
}
