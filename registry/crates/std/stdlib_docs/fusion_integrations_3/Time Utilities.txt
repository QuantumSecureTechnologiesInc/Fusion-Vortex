// fusion-std/src/time.rs
// Time utilities backed by the HyperRing

use std::time::Duration;
use fusion_rt_core::reactor::{Sleep, RingOp};

/// async sleep that yields to the reactor (non-blocking)
pub async fn sleep(duration: Duration) {
    let reactor = crate::rt();
    // Re-construct the Sleep primitive from rt-core
    // This connects the user API to the internal reactor logic
    Sleep::new(duration, reactor).await
}
