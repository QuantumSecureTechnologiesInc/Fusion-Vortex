// fusion-std/src/compute.rs
// Traits for Heterogeneous Compute

use fusion_rt_core::{Device, JoinHandle};
use std::future::Future;

/// Extension trait to make spawning tasks ergonomic
pub trait SpawnExt {
    fn spawn_local<F>(future: F) -> JoinHandle<F::Output>
    where F: Future + Send + 'static, F::Output: Send + 'static;

    fn spawn_gpu<F>(future: F) -> JoinHandle<F::Output>
    where F: Future + Send + 'static, F::Output: Send + 'static;
}

// We implement this on the 'Global' scope or unit type for convenience
pub struct Fusion;

impl Fusion {
    pub fn spawn<F>(future: F) -> JoinHandle<F::Output>
    where F: Future + Send + 'static, F::Output: Send + 'static
    {
        fusion_rt_core::spawn(future)
    }
}

// GPU Extension
pub trait GpuExt {
    fn to_gpu(&self) -> Self;
}

// Example impl for Vectors
impl GpuExt for Vec<f32> {
    fn to_gpu(&self) -> Self {
        // In reality, this allocates GPU memory via the reactor
        // and performs a DMA copy.
        self.clone() 
    }
}
