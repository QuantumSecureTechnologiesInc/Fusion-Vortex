//! src/core/telemetry.rs
//! Global context management.
//!
//! Ensures every story knows *where* and *when* it is happening.

use serde::{Serialize, Deserialize};
use std::env;

#[derive(Serialize, Deserialize, Debug, Clone)]
pub enum Environment {
    Development,
    Staging,
    Production,
    Research,
}

/// A snapshot of the runtime environment.
#[derive(Serialize, Deserialize, Debug, Clone)]
pub struct RuntimeContext {
    pub env: Environment,
    pub region: String,
    pub version: String,
}

impl RuntimeContext {
    /// Captures the current environment state.
    /// Reads from standard ENV vars (APP_ENV, REGION) with sensible defaults.
    pub fn current() -> Self {
        let env_str = env::var("APP_ENV").unwrap_or_else(|_| "Development".into());
        let env = match env_str.to_lowercase().as_str() {
            "production" | "prod" => Environment::Production,
            "staging" => Environment::Staging,
            "research" => Environment::Research,
            _ => Environment::Development,
        };

        Self {
            env,
            region: env::var("REGION").unwrap_or_else(|_| "local-sim".into()),
            version: env!("CARGO_PKG_VERSION").to_string(),
        }
    }
}
