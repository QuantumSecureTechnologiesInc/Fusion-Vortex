//! src/cycle/error.rs
//! Defines the narrative error types for the Compute (Cycle) layer.
//!
//! These errors describe why a compute agreement or task failed on the decentralised grid.

use thiserror::Error;

#[derive(Error, Debug)]
pub enum CycleFailure {
    #[error("Grid connection lost: {0}")]
    GridUnreachable(String),

    #[error("Contract breach: {reason} (Agreement ID: {agreement_id})")]
    ContractBreach {
        agreement_id: String,
        reason: String,
    },

    #[error("Market volatility: Bid rejected. Required: {required}, Offered: {offered}")]
    InsufficientBid {
        required: u64,
        offered: u64,
    },

    #[error("Resource exhaustion: Node {node_id} cannot fulfil request for {resource}")]
    ResourceUnavailable {
        node_id: String,
        resource: String,
    },

    #[error("Consensus failure: Nodes disagreed on result (Confidence: {confidence}%)")]
    ConsensusMismatch {
        confidence: f32,
    },
}
