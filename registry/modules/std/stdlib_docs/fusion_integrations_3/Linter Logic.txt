// flux/src/main.rs
// The Fusion Linter ("Flux") Entry Point - Updated with v1/v2 checks

mod rules;
mod diagnostics;

use clap::{Parser, Subcommand};
use colored::*;
use std::path::{Path, PathBuf};
use std::fs;
use anyhow::Result;
use crate::diagnostics::{Diagnostic, Level, DiagnosticReporter};
use crate::rules::LinterContext;

#[derive(Parser)]
#[command(name = "flux")]
#[command(about = "Fusion Architecture Linter", long_about = None)]
struct Cli {
    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    /// Check the project for architectural violations
    Check {
        /// Treat warnings as errors
        #[arg(long)]
        strict: bool,
    },
    /// Automatically fix simple violations
    Fix,
}

fn main() -> Result<()> {
    let cli = Cli::parse();
    
    // 1. Setup Context
    let root = std::env::current_dir()?;
    let manifest_path = root.join("Fusion.toml");
    
    if !manifest_path.exists() {
        eprintln!("{} Fusion.toml not found. Flux must be run in a Fusion project root.", "Fatal:".red().bold());
        std::process::exit(1);
    }

    let manifest_content = fs::read_to_string(&manifest_path)?;
    // We do a loose parse here just for linting purposes
    let manifest: rules::PartialManifest = toml::from_str(&manifest_content)
        .map_err(|e| anyhow::anyhow!("Failed to parse Fusion.toml: {}", e))?;

    let ctx = LinterContext {
        root: root.clone(),
        manifest,
        manifest_path,
    };

    let mut reporter = DiagnosticReporter::new();

    // 2. Execute Rules
    println!("{} Flux analysing {}...", "Running".blue().bold(), ctx.manifest.package.name.cyan());
    
    // General Rules
    rules::check_runtime_compatibility(&ctx, &mut reporter);
    rules::check_blocking_io(&ctx, &mut reporter);
    rules::check_ffi_safety(&ctx, &mut reporter);
    
    // Version Specific Rules
    rules::check_nebula_strictness(&ctx, &mut reporter); // v2.0 Checks
    rules::check_v1_native_purity(&ctx, &mut reporter);  // v1 Checks

    match cli.command {
        Commands::Check { strict } => {
            reporter.print_summary();
            if reporter.has_errors() || (strict && reporter.has_warnings()) {
                std::process::exit(1);
            }
        }
        Commands::Fix => {
            // Apply fixes
            let fixes = rules::apply_fixes(&ctx, &reporter)?;
            println!("{} Applied {} fixes.", "Success".green().bold(), fixes);
        }
    }

    Ok(())
}
