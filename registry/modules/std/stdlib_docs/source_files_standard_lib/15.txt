//! src/prelude.rs
//! Library-defined operations on Language Primitives.
//!
//! This module defines Extension Traits that allow standard Rust types
//! (like String) to behave as Fusion-aware objects.

use crate::seal::{NeuralSanitiser, SecurityViolation};

/// Extends `String` and `str` with NeuralSeal capabilities.
pub trait FusionString {
    /// Attempt to sanitise this string using the default NeuralSanitiser.
    fn sanitise(&self) -> Result<String, SecurityViolation>;
    
    /// Checks if the string looks like a valid system ID (simple heuristic).
    fn is_valid_id(&self) -> bool;
}

impl FusionString for str {
    fn sanitise(&self) -> Result<String, SecurityViolation> {
        // We instantiate a transient sanitiser for ergonomic calls.
        // In hot loops, use the explicit NeuralSanitiser struct instead.
        let sanitiser = NeuralSanitiser::new();
        sanitiser.cleanse(self)
    }

    fn is_valid_id(&self) -> bool {
        !self.is_empty() && self.chars().all(|c| c.is_alphanumeric() || c == '-' || c == '_')
    }
}

impl FusionString for String {
    fn sanitise(&self) -> Result<String, SecurityViolation> {
        self.as_str().sanitise()
    }

    fn is_valid_id(&self) -> bool {
        self.as_str().is_valid_id()
    }
}
