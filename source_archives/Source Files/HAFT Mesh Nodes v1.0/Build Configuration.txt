# Status: Build Configuration
# Purpose: Enforce static linking for bare-metal deployment.

[build]
# This flag forces the compiler to link the C runtime statically.
# It ensures the resulting binary has no external dependencies on libc.so,
# allowing it to run on any Linux distro (Alpine, Scratch, etc.).
rustflags = ["-C", "target-feature=+crt-static"]

[target.x86_64-unknown-linux-gnu]
# Use the high-performance LLVM linker
linker = "rust-lld"
