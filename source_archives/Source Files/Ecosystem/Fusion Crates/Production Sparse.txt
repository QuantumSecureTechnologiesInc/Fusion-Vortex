/// Production CSR Matrix Operations.
/// 
/// High-performance SpMV (Sparse Matrix-Vector Multiplication).

use crate::CsrMatrix; // From previous step
use fusion_core::types::tensor::Vector1D;
use fusion_core::traits::Numeric;
use fusion_core::FusionResult;

impl<T: Numeric + Copy> CsrMatrix<T> {
    /// Optimized SpMV: y = A * x
    /// Uses OpenMP-style parallelization pattern (iterators) if extended.
    pub fn matvec_prod(&self, x: &Vector1D<T>) -> FusionResult<Vector1D<T>> {
        if self.shape[1] != x.shape[0] {
            return Err(fusion_core::FusionError::ShapeMismatch {
                op: "SpMV".into(),
                lhs: self.shape.to_vec(),
                rhs: x.shape.to_vec(),
            });
        }

        let mut output_data = vec![T::zero(); self.shape[0]];

        // Iterate Rows
        for i in 0..self.shape[0] {
            let row_start = self.row_ptr[i];
            let row_end = self.row_ptr[i + 1];
            
            let mut sum = T::zero();
            
            // Iterate Non-Zero elements in row
            for ptr in row_start..row_end {
                let col_idx = self.col_indices[ptr];
                let val = self.values[ptr];
                let x_val = x.get([col_idx])?;
                
                sum = sum + (val * x_val);
            }
            output_data[i] = sum;
        }

        // Safe construction
        fusion_core::types::tensor::Tensor::new(output_data, [self.shape[0]])
    }
}
