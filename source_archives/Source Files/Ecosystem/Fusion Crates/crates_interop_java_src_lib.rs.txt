/// Production Java Bridge (JNI).
/// 
/// Features:
/// - Exception handling: Translates Rust Result/Panic to Java Exceptions.
/// - Resource management: Handles native pointers.

use fusion_core::types::tensor::{Tensor, Matrix};
use std::sync::{Arc, Mutex};
use std::ffi::{CStr, CString};
use libc::c_void;

// Mocking JNI types
pub type JNIEnv = *mut c_void;
pub type JClass = *mut c_void;
pub type JObject = *mut c_void;
pub type jint = i32;
pub type jdouble = f64;
pub type jlong = i64;

// Global map to hold Rust objects, referenced by ID (pointer address)
// In prod, use `jni::objects::GlobalRef` or store pointer in Java long field.
// Here we simulate the pointer-in-long approach safely.

/// Helper to cast raw pointer to Matrix reference.
unsafe fn get_matrix<'a>(ptr: jlong) -> &'a mut Matrix<f64> {
   assert!(ptr != 0, "Null pointer passed to native");
   &mut *(ptr as *mut Matrix<f64>)
}

/// Helper to throw Java exception.
unsafe fn throw_exception(env: JNIEnv, msg: &str) {
   // Call env->ThrowNew(...)
   println!("[JNI Error] Throwing Java Exception: {}", msg);
}

#[no_mangle]
pub extern "system" fn Java_com_fusion_Matrix_nativeCreate(
   _env: JNIEnv,
   _class: JClass,
   rows: jint,
   cols: jint
) -> jlong {
   // Catch panics to prevent crashing JVM
   let result = std::panic::catch_unwind(|| {
       let matrix = Tensor::zeros([rows as usize, cols as usize]);
       // Leak memory to Java (Java holds the pointer)
       let boxed = Box::new(matrix);
       Box::into_raw(boxed) as jlong
   });

   match result {
       Ok(ptr) => ptr,
       Err(_) => {
           // throw_exception(_env, "Panic in nativeCreate");
           0
       }
   }
}

#[no_mangle]
pub extern "system" fn Java_com_fusion_Matrix_nativeSet(
   env: JNIEnv,
   _class: JClass,
   ptr: jlong,
   r: jint,
   c: jint,
   val: jdouble
) {
   let result = std::panic::catch_unwind(|| {
       let matrix = unsafe { get_matrix(ptr) };
       matrix.set([r as usize, c as usize], val)
   });

   match result {
       Ok(Ok(_)) => {},
       Ok(Err(e)) => unsafe { throw_exception(env, &format!("{:?}", e)) },
       Err(_) => unsafe { throw_exception(env, "Panic in nativeSet") },
   }
}

#[no_mangle]
pub extern "system" fn Java_com_fusion_Matrix_nativeGet(
   env: JNIEnv,
   _class: JClass,
   ptr: jlong,
   r: jint,
   c: jint
) -> jdouble {
   let result = std::panic::catch_unwind(|| {
       let matrix = unsafe { get_matrix(ptr) };
       matrix.get([r as usize, c as usize])
   });

   match result {
       Ok(Ok(val)) => val,
       Ok(Err(e)) => {
           unsafe { throw_exception(env, &format!("{:?}", e)); }
           0.0
       },
       Err(_) => {
           unsafe { throw_exception(env, "Panic in nativeGet"); }
           0.0
       }
   }
}

#[no_mangle]
pub extern "system" fn Java_com_fusion_Matrix_nativeFree(
   _env: JNIEnv,
   _class: JClass,
   ptr: jlong
) {
   if ptr != 0 {
       // Retake ownership to drop
       unsafe {
           let _ = Box::from_raw(ptr as *mut Matrix<f64>);
       }
   }
}
