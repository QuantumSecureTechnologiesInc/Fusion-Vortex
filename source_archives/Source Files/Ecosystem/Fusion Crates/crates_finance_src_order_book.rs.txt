/// Production Limit Order Book.
/// 
/// Features:
/// - Decimal Arithmetic (rust_decimal) for currency safety (No Floats).
/// - Transactional Matching (Price-Time Priority).
/// - Precise Error Handling without panics.

use rust_decimal::Decimal;
use rust_decimal_macros::dec;
use std::collections::{BTreeMap, VecDeque};
use chrono::{DateTime, Utc};
use thiserror::Error;

#[derive(Error, Debug)]
pub enum OrderBookError {
   #[error("Invalid Price: {0}")]
   InvalidPrice(Decimal),
   #[error("Invalid Quantity: {0}")]
   InvalidQuantity(Decimal),
   #[error("Order not found: {0}")]
   OrderNotFound(u64),
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum Side { Buy, Sell }

#[derive(Debug, Clone)]
pub struct Order {
   pub id: u64,
   pub price: Decimal,
   pub quantity: Decimal,
   pub side: Side,
   pub timestamp: DateTime<Utc>,
}

#[derive(Debug)]
pub struct Trade {
   pub maker_id: u64,
   pub taker_id: u64,
   pub price: Decimal,
   pub quantity: Decimal,
}

pub struct OrderBook {
   // Price -> Queue of Orders.
   // Bids (Buy): Sorted Descending (Highest Price First).
   // Asks (Sell): Sorted Ascending (Lowest Price First).
   bids: BTreeMap<Decimal, VecDeque<Order>>, 
   asks: BTreeMap<Decimal, VecDeque<Order>>,
}

impl OrderBook {
   pub fn new() -> Self {
       Self {
           bids: BTreeMap::new(),
           asks: BTreeMap::new(),
       }
   }

   /// Process a limit order safely.
   pub fn place_limit_order(&mut self, mut order: Order) -> Result<Vec<Trade>, OrderBookError> {
       if order.price <= dec!(0) {
           return Err(OrderBookError::InvalidPrice(order.price));
       }
       if order.quantity <= dec!(0) {
           return Err(OrderBookError::InvalidQuantity(order.quantity));
       }

       let mut trades = Vec::new();

       if order.side == Side::Buy {
           self.match_buy_order(&mut order, &mut trades);
       } else {
           self.match_sell_order(&mut order, &mut trades);
       }

       // If not fully filled, rest on book (Maker)
       if order.quantity > dec!(0) {
           match order.side {
               Side::Buy => {
                   self.bids.entry(order.price).or_default().push_back(order);
               },
               Side::Sell => {
                   self.asks.entry(order.price).or_default().push_back(order);
               }
           }
       }

       Ok(trades)
   }

   fn match_buy_order(&mut self, order: &mut Order, trades: &mut Vec<Trade>) {
       let mut prices_to_remove = Vec::new();

       // Iterate through Ask levels from lowest to highest (Standard Iter)
       for (best_price, queue) in self.asks.iter_mut() {
           // Check crossing: If lowest ask > buy limit, no match possible
           if *best_price > order.price {
               break; 
           }

           // Match within this price level (Time Priority)
           while let Some(maker) = queue.front_mut() {
               if order.quantity <= dec!(0) { break; }

               let trade_qty = std::cmp::min(order.quantity, maker.quantity);
               let trade_price = maker.price;

               trades.push(Trade {
                   maker_id: maker.id,
                   taker_id: order.id,
                   price: trade_price,
                   quantity: trade_qty,
               });

               maker.quantity -= trade_qty;
               order.quantity -= trade_qty;

               if maker.quantity == dec!(0) {
                   queue.pop_front();
               } else {
                   // Maker filled remaining taker qty, taker done
                   break;
               }
           }

           if queue.is_empty() {
               prices_to_remove.push(*best_price);
           }
           
           if order.quantity <= dec!(0) { break; }
       }

       for p in prices_to_remove {
           self.asks.remove(&p);
       }
   }

   fn match_sell_order(&mut self, order: &mut Order, trades: &mut Vec<Trade>) {
       // Match against Bids (Highest Price First) -> Need rev iter
       // Since Rust BTreeMap doesn't allow mutation during iteration easily, 
       // we use a key collection loop approach.
       
       loop {
           // Peek best bid (Highest key)
           let best_price = match self.bids.keys().next_back() { 
               Some(p) => *p,
               None => break,
           };

           // Check crossing: If highest bid < sell limit, no match
           if best_price < order.price {
               break; 
           }

           let queue = self.bids.get_mut(&best_price).unwrap();
           let mut level_empty = false;
           
           while let Some(maker) = queue.front_mut() {
               if order.quantity <= dec!(0) { break; }

               let trade_qty = std::cmp::min(order.quantity, maker.quantity);
               let trade_price = maker.price; 

               trades.push(Trade {
                   maker_id: maker.id,
                   taker_id: order.id,
                   price: trade_price,
                   quantity: trade_qty,
               });

               maker.quantity -= trade_qty;
               order.quantity -= trade_qty;

               if maker.quantity == dec!(0) {
                   queue.pop_front();
               } else {
                   break;
               }
           }

           if queue.is_empty() {
               level_empty = true;
           }
           
           if level_empty {
               self.bids.remove(&best_price);
           }
           
           if order.quantity <= dec!(0) { break; }
       }
   }
}
