// Status: Foundation Layer
// Purpose: Unified error handling across the Fusion stack.

use std::fmt;
use std::io;

/// The Unified Fusion Error type.
///
/// This enum consolidates errors from the Kernel, Driver, and Service layers
/// into a single stack-traceable type. We avoid generic "Anyhow" types in the 
/// core kernel to ensure predictable failure modes for the Sentinel.
#[derive(Debug)]
pub enum FusionError {
    /// Low-level I/O failures (file not found, broken pipe).
    Io(io::Error),
    
    /// The HyperRing reactor has collapsed or panicked.
    ReactorStall(String),
    
    /// A cryptographic handshake failed (bad key or signature).
    CryptoFailure { reason: String, fatal: bool },
    
    /// The task queue exceeded backpressure limits.
    ExecutorOverload,
    
    /// WebAssembly runtime trap or linking error.
    WasmTrap(String),
    
    /// Distributed mesh consensus failure.
    ClusterSplitBrain,
}

impl std::error::Error for FusionError {}

impl fmt::Display for FusionError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Self::Io(e) => write!(f, "[IO] Kernel I/O fault: {}", e),
            Self::ReactorStall(msg) => write!(f, "[REACTOR] HyperRing Stalled: {}", msg),
            Self::CryptoFailure { reason, fatal } => {
                write!(f, "[CRYPTO] Handshake failed (Fatal: {}): {}", fatal, reason)
            }
            Self::ExecutorOverload => write!(f, "[EXEC] Task queue overflow - backpressure active"),
            Self::WasmTrap(msg) => write!(f, "[WASM] Sandbox trap: {}", msg),
            Self::ClusterSplitBrain => write!(f, "[CLUSTER] Mesh consensus lost (Split Brain)"),
        }
    }
}

impl From<io::Error> for FusionError {
    fn from(err: io::Error) -> Self {
        FusionError::Io(err)
    }
}

/// A specialized Result type for Fusion operations.
pub type Result<T> = std::result::Result<T, FusionError>;
