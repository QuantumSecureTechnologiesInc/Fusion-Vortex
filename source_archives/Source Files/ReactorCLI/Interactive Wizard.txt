// reactor_cli/src/interactive.rs
use clap::CommandFactory;
use inquire::{Text, Confirm};
use colored::*;

/// Automatically prompt for missing fields
/// `is_proxy`: If true, simulates v2.1 Host Bridge calls
pub fn wizard<T>(is_proxy: bool) -> Result<T, String> 
where T: clap::Parser + clap::CommandFactory + clap::FromArgMatches
{
    let cmd = T::command();
    
    if is_proxy {
        println!("{}", "--- Host Bridge Input Request ---".magenta());
    } else {
        println!("{}", "--- Fusion Setup Wizard ---".cyan());
    }
    
    for arg in cmd.get_arguments() {
        if arg.is_required_set() {
            let id = arg.get_id();
            let help = arg.get_help().unwrap_or("No description");
            
            // In v2.1 Proxy mode, we wouldn't use `inquire` directly (as it needs raw terminal),
            // we would serialize the request and send it to the host.
            // This code block simulates the user experience of that success.
            if is_proxy {
                 println!("   [HOST] Please provide value for '{}' ({}):", id, help);
                 // Simulate host return
                 println!("   < Received 'mock_value' from Host >");
            } else {
                let _value = Text::new(id.as_str())
                    .with_help_message(help)
                    .with_placeholder("Enter value...")
                    .prompt()
                    .map_err(|e| e.to_string())?;
            }
        }
    }

    Err("Wizard complete. (Mock implementation)".to_string())
}
