/// TermBlink-GPU: Production Ready Terminal Browser
#[derive(Parser, Debug)]
#[command(author, version, about, long_about = None)]
struct Args {
    /// The URL to navigate to
    #[arg(default_value = "https://webgpu.github.io/webgpu-samples/")]
    url: String,

    /// Viewport width (internal rendering resolution)
    #[arg(long, default_value_t = 1280)]
    width: u32,

    /// Viewport height (internal rendering resolution)
    #[arg(long, default_value_t = 720)]
    height: u32,

    /// Target Frame Rate
    #[arg(long, default_value_t = 15)]
    fps: u64,

    /// Image Quality (1-100)
    #[arg(long, default_value_t = 50)]
    quality: i64,
}

/// A RAII guard to ensure the terminal is restored to a usable state
struct TerminalGuard;

impl TerminalGuard {
    fn new() -> Result<Self> {
        terminal::enable_raw_mode().context("Failed to enable raw mode")?;
        let mut stdout = stdout();
        stdout.execute(terminal::EnterAlternateScreen)?;
        stdout.execute(cursor::Hide)?;
        stdout.execute(event::EnableMouseCapture)?;
        Ok(Self)
    }
}

impl Drop for TerminalGuard {
    fn drop(&mut self) {
        let _ = terminal::disable_raw_mode();
        let mut stdout = stdout();
        let _ = stdout.execute(event::DisableMouseCapture);
        let _ = stdout.execute(cursor::Show);
        let _ = stdout.execute(terminal::LeaveAlternateScreen);
        println!(); // Clean prompt
    }
}

#[tokio::main]
async fn main() -> Result<()> {
    // --- 1. Setup Logging ---
    let file_appender = tracing_appender::rolling::daily("logs", "termblink.log");
    let (non_blocking, _guard) = tracing_appender::non_blocking(file_appender);
    tracing_subscriber::fmt()
        .with_writer(non_blocking)
        .with_ansi(false)
        .init();

    info!("Starting TermBlink-GPU...");
    let args = Args::parse();

    // --- 2. Terminal Setup (RAII) ---
    let _term_guard = TerminalGuard::new()?;

    // --- 3. Browser Setup ---
    let chrome_path = find_chrome_executable();
    
    // Persistent Profile
    let user_data_dir = std::env::current_dir()?.join("chrome_profile");
    fs::create_dir_all(&user_data_dir).context("Failed to create user data directory")?;

    let mut chrome_args = vec![
        "--headless=new".to_string(),
        "--no-sandbox".to_string(),
        "--enable-unsafe-webgpu".to_string(),
        "--disable-gpu-sandbox".to_string(),
        "--disable-infobars".to_string(),
        "--disable-blink-features=AutomationControlled".to_string(),
        format!("--window-size={},{}", args.width, args.height),
        format!("--user-data-dir={}", user_data_dir.to_string_lossy()),
    ];

    #[cfg(target_os = "linux")]
    {
        chrome_args.push("--use-angle=vulkan".to_string());
    }

    let mut builder = BrowserConfig::builder()
        .with_head()
        .args(chrome_args);

    if let Some(path) = chrome_path {
        builder = builder.chrome_executable(path);
    }

    let config = builder.build().context("Failed to build browser config")?;
    let (mut browser, mut handler) = Browser::launch(config)
        .await
        .context("Failed to launch browser. Ensure Chrome is installed via setup scripts.")?;

    let _handle = tokio::task::spawn(async move {
        while let Some(h) = handler.next().await {
            if h.is_err() { break; }
        }
    });

    // --- 4. Navigation ---
    info!("Navigating to {}", args.url);
    let page = browser.new_page(&args.url).await?;
    page.wait_for_navigation().await?;

    // --- 5. Event Loop ---
    let render_duration = Duration::from_millis(1000 / args.fps);
    let mut render_interval = time::interval(render_duration);
    let mut event_stream = EventStream::new();
    let mut last_render = Instant::now();

    loop {
        tokio::select! {
            _ = render_interval.tick() => {
                if last_render.elapsed() < render_duration.mul_f32(0.8) { continue; }

                let screenshot_res = page
                    .screenshot(
                        chromiumoxide::page::ScreenshotParams::builder()
                            .format(CaptureScreenshotFormat::Jpeg)
                            .quality(args.quality)
                            .build(),
                    )
                    .await;

                match screenshot_res {
                    Ok(bytes) => {
                        // CPU Offload
                        let (term_cols, term_rows) = terminal::size().unwrap_or((80, 24));
                        let processed = tokio::task::spawn_blocking(move || {
                            if let Ok(img) = image::load_from_memory(&bytes) {
                                let resized = img.resize_exact(
                                    term_cols as u32, 
                                    term_rows as u32 * 2, 
                                    FilterType::Nearest
                                );
                                Some((resized, term_cols, term_rows))
                            } else { None }
                        }).await?;

                        if let Some((img, cols, rows)) = processed {
                            let mut stdout = stdout();
                            render_frame(&img, cols, rows, &mut stdout)?;
                            last_render = Instant::now();
                        }
                    }
                    Err(_) => break,
                }
            }

            Some(event_res) = event_stream.next() => {
                if let Ok(event) = event_res {
                    match event {
                        Event::Key(key) => {
                            if (key.code == KeyCode::Char('c') && key.modifiers.contains(KeyModifiers::CONTROL)) 
                               || key.code == KeyCode::Esc {
                                break;
                            }
                            handle_key_event(&page, key).await.ok();
                        }
                        Event::Mouse(mouse) => {
                            let (cols, rows) = terminal::size().unwrap_or((80, 24));
                            handle_mouse_event(&page, mouse, cols, rows, args.width, args.height).await.ok();
                        }
                        _ => {}
                    }
                }
            }
        }
    }
    Ok(())
}

fn render_frame(img: &image::DynamicImage, cols: u16, rows: u16, stdout: &mut std::io::Stdout) -> Result<()> {
    stdout.queue(cursor::MoveTo(0, 0))?;
    for y in 0..rows {
        for x in 0..cols {
            let p = img.get_pixel(x as u32, y as u32 * 2);
            stdout.queue(style::SetBackgroundColor(Color::Rgb { r: p[0], g: p[1], b: p[2] }))?;
            stdout.queue(Print(" "))?;
        }
        stdout.queue(style::ResetColor)?;
        if y < rows - 1 { stdout.queue(Print("\r\n"))?; }
    }
    stdout.flush()?;
    Ok(())
}

async fn handle_key_event(page: &chromiumoxide::Page, key: event::KeyEvent) -> Result<()> {
    match key.code {
        KeyCode::Char(c) => {
             page.evaluate(format!("document.activeElement.value += '{}'; document.activeElement.dispatchEvent(new Event('input', {{bubbles:true}}));", c)).await?;
        },
        KeyCode::Backspace => {
             page.evaluate("document.activeElement.value = document.activeElement.value.slice(0, -1); document.activeElement.dispatchEvent(new Event('input', {{bubbles:true}}));").await?;
        },
        KeyCode::Enter => {
             page.evaluate("let e=new KeyboardEvent('keydown',{key:'Enter',code:'Enter',bubbles:true});document.activeElement.dispatchEvent(e);if(document.activeElement.form)document.activeElement.form.submit();").await?;
        },
        KeyCode::Down => { page.evaluate("window.scrollBy(0, 100);").await?; },
        KeyCode::Up => { page.evaluate("window.scrollBy(0, -100);").await?; },
        _ => {}
    }
    Ok(())
}

async fn handle_mouse_event(page: &chromiumoxide::Page, mouse: event::MouseEvent, term_cols: u16, term_rows: u16, vw: u32, vh: u32) -> Result<()> {
    let sx = vw as f64 / term_cols as f64;
    let sy = vh as f64 / term_rows as f64;
    let bx = mouse.column as f64 * sx;
    let by = mouse.row as f64 * sy;

    match mouse.kind {
        MouseEventKind::Down(b) => {
            page.execute(DispatchMouseEventParams::builder().r#type(DispatchMouseEventType::MousePressed).x(bx).y(by).button(map_btn(b)).click_count(1).build()).await?;
        }
        MouseEventKind::Up(b) => {
            page.execute(DispatchMouseEventParams::builder().r#type(DispatchMouseEventType::MouseReleased).x(bx).y(by).button(map_btn(b)).click_count(1).build()).await?;
        }
        MouseEventKind::Drag(_) | MouseEventKind::Moved => {
            page.execute(DispatchMouseEventParams::builder().r#type(DispatchMouseEventType::MouseMoved).x(bx).y(by).build()).await?;
        }
        MouseEventKind::ScrollDown => { page.evaluate("window.scrollBy(0, 100);").await?; },
        MouseEventKind::ScrollUp => { page.evaluate("window.scrollBy(0, -100);").await?; },
        _ => {}
    }
    Ok(())
}

fn map_btn(b: event::MouseButton) -> MouseButton {
    match b {
        event::MouseButton::Right => MouseButton::Right,
        event::MouseButton::Middle => MouseButton::Middle,
        _ => MouseButton::Left,
    }
}

fn find_chrome_executable() -> Option<PathBuf> {
    if let Ok(p) = std::env::var("CHROME_PATH") { if Path::new(&p).exists() { return Some(PathBuf::from(p)); } }
    let bin = if cfg!(target_os="macos") { Path::new("./bin/chrome-mac-x64/Google Chrome for Testing.app/Contents/MacOS/Google Chrome for Testing") } 
              else if cfg!(target_os="windows") { Path::new("./bin/chrome-win64/chrome.exe") } 
              else { Path::new("./bin/chrome-linux64/chrome") };
    if bin.exists() { std::fs::canonicalize(bin).ok() } else { None }
}
