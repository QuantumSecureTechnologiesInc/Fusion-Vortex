/**
 * Fusion Core Test Suite
 * ----------------------
 * Verifies the integrity of the Fusion Standard utilities.
 * Focus: Hybrid Key Validation, Result Monad safety, and Deep Merge security.
 *
 * Usage: npx jest fusion-core.test.ts
 */

import { Guard } from './guard';
import { Result, ok, err, map } from './result';
import { Task } from './task';
import { deepMerge } from './deepMerge';

// Mock Telemetry for Task testing
jest.mock('./telemetry', () => ({
  Telemetry: {
    measureAsync: async (_name: string, fn: (span: any) => Promise<any>) => {
      const mockSpan = { addEvent: jest.fn(), end: jest.fn() };
      return fn(mockSpan);
    }
  }
}));

describe('Fusion Core Ecosystem', () => {

  describe('Guard (Hybrid Security)', () => {
    it('should REJECT a bundle missing the Quantum component', () => {
      const classicalOnly = {
        classical: { kty: 'EC', crv: 'P-256' },
        // Quantum missing
      };
      expect(Guard.isHybridKeyBundle(classicalOnly)).toBe(false);
    });

    it('should REJECT a bundle missing the Classical component', () => {
      const quantumOnly = {
        // Classical missing
        quantum: { matrixA: [], vectorT: [] }
      };
      expect(Guard.isHybridKeyBundle(quantumOnly)).toBe(false);
    });

    it('should ACCEPT a valid 50/50 Hybrid Bundle', () => {
      const validHybrid = {
        classical: { kty: 'EC', crv: 'P-256' },
        quantum: { matrixA: [[1]], vectorT: [1] } // Minimal valid shape
      };
      expect(Guard.isHybridKeyBundle(validHybrid)).toBe(true);
    });

    it('should throw explicit errors for byte length mismatches', () => {
      const buffer = new Uint8Array(16);
      expect(() => Guard.byteLength(buffer, 32, 'SessionKey'))
        .toThrow(/Guard: 'SessionKey' must be 32 bytes/);
    });
  });

  describe('Result Monad (Safety)', () => {
    it('should allow mapping over success values', () => {
      const success: Result<number, string> = ok(10);
      const doubled = map(success, (x) => x * 2);
      
      expect(doubled.isSuccess()).toBe(true);
      expect(doubled.unwrap()).toBe(20);
    });

    it('should skip mapping if result is an error', () => {
      const failure: Result<number, string> = err("Original Error");
      const skipped = map(failure, (x) => x * 2);

      expect(skipped.isFailure()).toBe(true);
      expect(skipped.unwrapErr()).toBe("Original Error");
    });

    it('should prevent unwrapping an error', () => {
      const failure = err("Boom");
      expect(() => failure.unwrap()).toThrow("Boom");
    });
  });

  describe('Task (Observability)', () => {
    it('should capture success without throwing', async () => {
      const task = Task.from("TestTask", async () => "Success");
      const result = await task.run();

      expect(result.isSuccess()).toBe(true);
      expect(result.unwrap()).toBe("Success");
    });

    it('should capture exceptions as Result.Err', async () => {
      const task = Task.from("FailTask", async () => {
        throw new Error("Network Down");
      });
      const result = await task.run();

      expect(result.isFailure()).toBe(true);
      expect(result.unwrapErr()).toBeInstanceOf(Error);
      expect((result.unwrapErr() as Error).message).toBe("Network Down");
    });
  });

  describe('Deep Merge (Prototype Protection)', () => {
    it('should merge nested objects correctly', () => {
      const target = { a: 1, b: { x: 10 } };
      const source = { b: { y: 20 }, c: 3 };
      const merged = deepMerge(target, source);

      expect(merged).toEqual({ a: 1, b: { x: 10, y: 20 }, c: 3 });
    });

    it('should IGNORE prototype pollution attempts', () => {
      const target = {};
      const maliciousPayload = JSON.parse('{"__proto__": {"polluted": true}}');
      
      const merged = deepMerge(target, maliciousPayload);
      
      // @ts-ignore - Verification
      expect({}.polluted).toBeUndefined();
      expect(merged.__proto__).not.toHaveProperty('polluted');
    });
  });

});
