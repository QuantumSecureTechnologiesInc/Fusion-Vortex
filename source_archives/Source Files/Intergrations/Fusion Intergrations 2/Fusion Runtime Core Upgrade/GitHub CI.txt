name: Fusion CI
on: [push, pull_request]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Install Rust
        uses: dtolnay/rust-toolchain@nightly
      - name: Test (Simulation)
        run: cargo test --workspace --features simulation
      - name: Build Release
        run: cargo build --release --package fusion_runtime_core
