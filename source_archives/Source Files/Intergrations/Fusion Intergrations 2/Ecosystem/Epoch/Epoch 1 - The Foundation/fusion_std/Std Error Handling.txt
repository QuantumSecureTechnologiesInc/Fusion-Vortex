/// Standard Library Error System.
/// Bridges low-level OS/Kernel errors into user-facing Fusion errors.

use thiserror::Error;
use fusion_core::FusionError;
use std::io;

#[derive(Error, Debug)]
pub enum StdError {
    #[error("Fusion Core Error: {0}")]
    Core(#[from] FusionError),

    #[error("I/O Error: {0}")]
    Io(#[from] io::Error),

    #[error("Filesystem Security Violation: Path '{0}' escapes sandbox")]
    PathEscapesSandbox(String),

    #[error("Permission Denied: {0}")]
    PermissionDenied(String),

    #[error("Serialization Error: {0}")]
    Serialization(String),
}

pub type StdResult<T> = Result<T, StdError>;
