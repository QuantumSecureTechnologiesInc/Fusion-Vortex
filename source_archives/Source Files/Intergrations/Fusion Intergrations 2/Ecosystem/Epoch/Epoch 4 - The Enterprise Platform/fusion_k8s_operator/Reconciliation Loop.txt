/// The Brain of the Operator.
/// Reconciles the state of FusionApp resources.

use crate::crd::{FusionApp, FusionAppStatus};
use kube::{Api, Client, ResourceExt};
use kube::runtime::controller::{Action, Controller};
use std::sync::Arc;
use tokio::time::Duration;
use futures::StreamExt;

pub struct Context {
    client: Client,
}

/// Entry point to start the controller manager.
pub async fn run() -> anyhow::Result<()> {
    let client = Client::try_default().await?;
    let context = Arc::new(Context { client: client.clone() });
    
    let fusion_apps = Api::<FusionApp>::all(client);
    
    Controller::new(fusion_apps, kube::runtime::watcher::Config::default())
        .run(reconcile, error_policy, context)
        .for_each(|_| futures::future::ready(()))
        .await;
        
    Ok(())
}

/// Error policy for the controller.
fn error_policy(_app: Arc<FusionApp>, _error: &kube::Error, _ctx: Arc<Context>) -> Action {
