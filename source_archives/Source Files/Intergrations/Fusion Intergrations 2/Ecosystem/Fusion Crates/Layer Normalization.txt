/// Production Layer Normalization.
/// 
/// Normalizes inputs across the feature dimension (usually last dimension).
/// Features: Numerical stability via epsilon (eps).

use fusion_ai_core::{Variable, Linear, Layer};
use fusion_core::types::tensor::{Tensor, Matrix};
use fusion_core::FusionResult;

pub struct LayerNorm {
    pub gamma: Variable, // Scale parameter (trainable)
    pub beta: Variable,  // Bias parameter (trainable)
    pub eps: f64,
}

impl LayerNorm {
    pub fn new(normalized_shape: usize, eps: f64) -> Self {
        Self {
            gamma: Variable::new(Tensor::ones([normalized_shape])),
            beta: Variable::new(Tensor::zeros([normalized_shape])),
            eps,
        }
    }
}

impl Layer for LayerNorm {
    /// Forward Pass: y = (x - mean) / sqrt(variance + eps) * gamma + beta
    fn forward(&self, x: Variable) -> Variable {
        // Assuming x is Rank 2 [Batch, Features]
        let (n_samples, dim) = (x.data.shape[0], x.data.shape[1]);
        let mut output = Tensor::zeros([n_samples, dim]); // Final normalized tensor

        for i in 0..n_samples {
            // Calculate Mean and Variance for the i-th sample (row)
            let mut mean = 0.0;
            let mut var = 0.0;

            for j in 0..dim { mean += x.data.get([i, j]).unwrap(); }
            mean /= dim as f64;

            for j in 0..dim {
                let diff = x.data.get([i, j]).unwrap() - mean;
                var += diff * diff;
            }
            var /= dim as f64;

            let std_inv = (var + self.eps).sqrt().recip();

            // Apply normalization
            for j in 0..dim {
                let val = x.data.get([i, j]).unwrap();
                let normalized = (val - mean) * std_inv;
                let final_val = normalized * self.gamma.data.get([j]).unwrap() + self.beta.data.get([j]).unwrap();
                
                output.set([i, j], final_val).unwrap();
            }
        }

        Variable::new(output)
    }

    fn parameters(&self) -> Vec<Variable> {
        vec![self.gamma.clone(), self.beta.clone()]
    }
}
