#include "hc_gpu_universal.h"
#include "hc_math_core.h"
#include <cuda_runtime.h>
#include <stdio.h>

#define BLOCK_SIZE 256

__global__ void hc_cuda_entropy_kernel(uint64_t seed_base, uint64_t blinding_seed, uint8_t* out_buffer, size_t count) {
    uint64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= count) return;

    uint8_t local_key[32];
    hc_generate_single_key(seed_base, blinding_seed, idx, local_key);

    uint8_t* target = out_buffer + (idx * 32);
    #pragma unroll
    for(int i=0; i<32; i++) {
        target[i] = local_key[i];
    }
}

extern "C" {

hc_gpu_status_t hc_cuda_initialize(void) {
    int deviceCount = 0;
    if (cudaGetDeviceCount(&deviceCount) != cudaSuccess || deviceCount == 0) {
        return HC_GPU_ERR_NO_DEVICE;
    }
    cudaSetDevice(0);
    return HC_GPU_SUCCESS;
}

void hc_cuda_shutdown(void) {
    cudaDeviceReset();
}

hc_gpu_status_t hc_cuda_generate_entropy_batch(uint64_t seed_base, uint64_t blinding_seed, uint8_t* out_buffer, size_t count) {
    uint8_t *d_buffer = NULL;
    size_t total_bytes = count * 32;

    if (cudaMalloc(&d_buffer, total_bytes) != cudaSuccess) return HC_GPU_ERR_MEMORY;

    int grid_size = (int)((count + BLOCK_SIZE - 1) / BLOCK_SIZE);
    hc_cuda_entropy_kernel<<<grid_size, BLOCK_SIZE>>>(seed_base, blinding_seed, d_buffer, count);

    if (cudaGetLastError() != cudaSuccess) {
        cudaFree(d_buffer);
        return HC_GPU_ERR_KERNEL_FAILURE;
    }

    if (cudaDeviceSynchronize() != cudaSuccess) {
        cudaFree(d_buffer);
        return HC_GPU_ERR_KERNEL_FAILURE;
    }

    if (cudaMemcpy(out_buffer, d_buffer, total_bytes, cudaMemcpyDeviceToHost) != cudaSuccess) {
        cudaFree(d_buffer);
        return HC_GPU_ERR_MEMORY;
    }

    cudaFree(d_buffer);
    return HC_GPU_SUCCESS;
}

} // extern "C"
