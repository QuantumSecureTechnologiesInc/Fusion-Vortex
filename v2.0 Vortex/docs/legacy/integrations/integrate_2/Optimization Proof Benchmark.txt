/*
 * HyperCycle v1.1 Origin - Optimization Verification Benchmark
 * ============================================================
 *
 * This benchmark verifies the efficiency gains from the "Optimization Document"
 * techniques integrated into the PQC Boost Suite.
 *
 * 1. "Combine Rotations": Compares sequential rotation vs. pre-composed rotation.
 * 2. "Quaternion Arithmetic": Measures raw throughput of the fixed-point engine.
 *
 * Compile: gcc -O3 hc_opt_benchmark.c -o hc_bench
 */

#include <stdio.h>
#include <time.h>
#include "hc_math_boost.h"

// High-resolution timer
double get_time_sec() {
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return ts.tv_sec + ts.tv_nsec * 1e-9;
}

int main() {
    printf("========================================================\n");
    printf("  HyperCycle v1.1 Origin - Optimization Benchmark\n");
    printf("========================================================\n");

    const int ITERATIONS = 5000000;
    
    // Test Vectors
    hc_quat_t rot1 = {HC_SCALE, 100, 200, 300}; 
    hc_quat_t rot2 = {HC_SCALE, 50, -50, 50};
    hc_quat_t vec  = {0, 1000, 0, 0};
    hc_quat_t result;

    printf("[Test 1] Rotation Combination Efficiency\n");
    printf("  Iterations: %d\n", ITERATIONS);

    // --- Case A: Sequential Rotations ---
    // v' = Rot2( Rot1( v ) )
    double start_seq = get_time_sec();
    for(int i=0; i<ITERATIONS; i++) {
        hc_quat_t temp;
        hc_quat_rotate(&rot1, &vec, &temp);
        hc_quat_rotate(&rot2, &temp, &result);
        
        // Prevent compiler optimization
        vec.w ^= result.w & 1; 
    }
    double time_seq = get_time_sec() - start_seq;
    printf("  Sequential Time: %.6f sec\n", time_seq);

    // --- Case B: Combined Rotations (Optimized) ---
    // Rot_combined = Rot2 * Rot1
    // v' = Combined( v )
    double start_comb = get_time_sec();
    
    // One-time cost (Simulating session setup)
    hc_quat_t combined;
    hc_quat_compose_rotations(&rot2, &rot1, &combined);
    
    for(int i=0; i<ITERATIONS; i++) {
        hc_quat_rotate(&combined, &vec, &result);
        vec.w ^= result.w & 1;
    }
    double time_comb = get_time_sec() - start_comb;
    printf("  Combined Time:   %.6f sec\n", time_comb);

    // --- Analysis ---
    double speedup = time_seq / time_comb;
    printf("  Speedup Factor:  %.2fx\n", speedup);
    
    if (speedup > 1.2) {
        printf("[PASS] Optimization verified. Significant performance gain.\n");
    } else {
        printf("[WARN] Optimization gain marginal (Check pipeline efficiency).\n");
    }

    // --- Gimbal Lock Check ---
    // Mathematically implicit in quaternion structure, but we verify 
    // behavior at 90-degree pitch equivalent.
    // (Omitted for brevity as it's a mathematical property, not performance).

    return 0;
}
