#include "hc_gpu_universal.h"
#include "hc_math_core.h"
#include <hip/hip_runtime.h>
#include <stdio.h>

#define BLOCK_SIZE 256

__global__ void hc_hip_entropy_kernel(uint64_t seed_base, uint64_t blinding_seed, uint8_t* out_buffer, size_t count) {
    uint64_t idx = hipBlockIdx_x * hipBlockDim_x + hipThreadIdx_x;
    if (idx >= count) return;

    uint8_t local_key[32];
    hc_generate_single_key(seed_base, blinding_seed, idx, local_key);

    uint8_t* target = out_buffer + (idx * 32);
    #pragma unroll
    for(int i=0; i<32; i++) {
        target[i] = local_key[i];
    }
}

extern "C" {

hc_gpu_status_t hc_rocm_initialize(void) {
    int deviceCount = 0;
    if (hipGetDeviceCount(&deviceCount) != hipSuccess || deviceCount == 0) {
        return HC_GPU_ERR_NO_DEVICE;
    }
    hipSetDevice(0);
    return HC_GPU_SUCCESS;
}

void hc_rocm_shutdown(void) {
    hipDeviceReset();
}

hc_gpu_status_t hc_rocm_generate_entropy_batch(uint64_t seed_base, uint64_t blinding_seed, uint8_t* out_buffer, size_t count) {
    uint8_t *d_buffer = NULL;
    size_t total_bytes = count * 32;

    if (hipMalloc(&d_buffer, total_bytes) != hipSuccess) return HC_GPU_ERR_MEMORY;

    int grid_size = (int)((count + BLOCK_SIZE - 1) / BLOCK_SIZE);
    hipLaunchKernelGGL(hc_hip_entropy_kernel, dim3(grid_size), dim3(BLOCK_SIZE), 0, 0, seed_base, blinding_seed, d_buffer, count);

    if (hipGetLastError() != hipSuccess) {
        hipFree(d_buffer);
        return HC_GPU_ERR_KERNEL_FAILURE;
    }
    
    if (hipDeviceSynchronize() != hipSuccess) {
        hipFree(d_buffer);
        return HC_GPU_ERR_KERNEL_FAILURE;
    }

    if (hipMemcpy(out_buffer, d_buffer, total_bytes, hipMemcpyDeviceToHost) != hipSuccess) {
        hipFree(d_buffer);
        return HC_GPU_ERR_MEMORY;
    }

    hipFree(d_buffer);
    return HC_GPU_SUCCESS;
}

} // extern "C"
