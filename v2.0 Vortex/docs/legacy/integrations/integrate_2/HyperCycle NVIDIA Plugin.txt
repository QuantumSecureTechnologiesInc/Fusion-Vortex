#define HC_CUDA_BLOCK_SIZE 256
#define HC_CQC_CYCLES 97
#define HC_WARP_SIZE 32

// --- Error Handling Macro ---
#define HC_CUDA_CHECK(call) \
    do { \
        cudaError_t err = call; \
        if (err != cudaSuccess) { \
            fprintf(stderr, "[HyperCycle CUDA Error] %s:%d: %s\n", __FILE__, __LINE__, cudaGetErrorString(err)); \
            return -1; \
        } \
    } while (0)

// --- Device: Heisenberg-Euler Chaos Primitive ---
__device__ __forceinline__ uint64_t heisenberg_euler_step(uint64_t state, uint64_t neighbor) {
    uint64_t next = state * 6364136223846793005ULL + 1442695040888963407ULL;
    uint64_t non_linear = (state >> 33) ^ (state << 11);
    next ^= non_linear;
    next ^= (neighbor >> 19) ^ (neighbor << 5);
    return next;
}

__global__ void 
__launch_bounds__(HC_CUDA_BLOCK_SIZE)
hc_cqc_keygen_kernel_cuda(uint64_t* seeds, uint8_t* pk_out, uint8_t* sk_out, int num_keys) {
    int global_tid = blockIdx.x * blockDim.x + threadIdx.x;
    int lane_id = threadIdx.x % HC_WARP_SIZE;
    int warp_id = global_tid / HC_WARP_SIZE; // Effectively the Key ID

    if (warp_id >= num_keys) return;

    // Load Initial Entropy (Two trajectories per thread)
    const int key_offset = warp_id * 64;
    
    // Boundary check for memory safety
    if (key_offset + lane_id + 32 >= num_keys * 64) return; // Should not happen if grid is correct, but safe.

    uint64_t s0 = seeds[key_offset + lane_id];
    uint64_t s1 = seeds[key_offset + lane_id + 32];

    s0 ^= 0xBADC0FFEE0DDF00DULL;
    s1 ^= 0xDEADBEEFCAFEBABEULL;

    uint64_t* sk_ptr = (uint64_t*)sk_out;
    sk_ptr[key_offset + lane_id] = s0;
    sk_ptr[key_offset + lane_id + 32] = s1;

    uint64_t p0 = s0;
    uint64_t p1 = s1;

    #pragma unroll
    for (int i = 0; i < HC_CQC_CYCLES; i++) {
        // Intra-Warp Mixing (Horizontal)
        int stride = 1 << (i % 5);
        uint64_t n0 = __shfl_xor_sync(0xFFFFFFFF, p0, stride);
        uint64_t n1 = __shfl_xor_sync(0xFFFFFFFF, p1, stride);

        // Inter-Trajectory Mixing (Vertical)
        if (i % 3 == 0) {
            uint64_t temp = n0;
            n0 ^= p1;
            n1 ^= p0; 
        }

        p0 = heisenberg_euler_step(p0, n0);
        p1 = heisenberg_euler_step(p1, n1);
    }

    uint64_t* pk_ptr = (uint64_t*)pk_out;
    pk_ptr[key_offset + lane_id] = p0;
    pk_ptr[key_offset + lane_id + 32] = p1;
}

typedef struct {
    uint64_t* d_seeds;
    uint8_t* d_pk;
    uint8_t* d_sk;
    uint64_t* h_pinned_seeds;
    uint8_t* h_pinned_pk;
    uint8_t* h_pinned_sk;
    cudaStream_t stream;
    size_t    key_capacity;
} hc_cuda_context_t;

static hc_cuda_context_t g_ctx = {0};

extern "C" {

static int cuda_cqc_keypair_impl(uint64_t* seeds, uint8_t* pk_buffer, uint8_t* sk_buffer, size_t count) {
    if (seeds == NULL || pk_buffer == NULL || sk_buffer == NULL) return -2;

    if (count > g_ctx.key_capacity) {
        if (g_ctx.d_seeds) {
            cudaFree(g_ctx.d_seeds); 
            cudaFree(g_ctx.d_pk); 
            cudaFree(g_ctx.d_sk);
            cudaFreeHost(g_ctx.h_pinned_seeds);
            cudaFreeHost(g_ctx.h_pinned_pk);
            cudaFreeHost(g_ctx.h_pinned_sk);
            cudaStreamDestroy(g_ctx.stream);
        }

        size_t total_bytes = count * 512;
        // Allocate with Error Checking
        HC_CUDA_CHECK(cudaMalloc(&g_ctx.d_seeds, total_bytes));
        HC_CUDA_CHECK(cudaMalloc(&g_ctx.d_pk, total_bytes));
        HC_CUDA_CHECK(cudaMalloc(&g_ctx.d_sk, total_bytes));
        
        // Use Pinned Memory for Speed
        HC_CUDA_CHECK(cudaMallocHost(&g_ctx.h_pinned_seeds, total_bytes));
        HC_CUDA_CHECK(cudaMallocHost(&g_ctx.h_pinned_pk, total_bytes));
        HC_CUDA_CHECK(cudaMallocHost(&g_ctx.h_pinned_sk, total_bytes));
        
        HC_CUDA_CHECK(cudaStreamCreate(&g_ctx.stream));
        g_ctx.key_capacity = count;
    }

    size_t total_bytes = count * 512;
    memcpy(g_ctx.h_pinned_seeds, seeds, total_bytes);

    HC_CUDA_CHECK(cudaMemcpyAsync(g_ctx.d_seeds, g_ctx.h_pinned_seeds, total_bytes, cudaMemcpyHostToDevice, g_ctx.stream));

    // 1 Warp = 1 Key. Block = 256 threads = 8 Warps = 8 Keys.
    int blocks = (count + 7) / 8;
    hc_cqc_keygen_kernel_cuda<<<blocks, HC_CUDA_BLOCK_SIZE, 0, g_ctx.stream>>>(
        g_ctx.d_seeds, g_ctx.d_pk, g_ctx.d_sk, (int)count
    );

    cudaError_t kernelErr = cudaGetLastError();
    if (kernelErr != cudaSuccess) {
        fprintf(stderr, "[HyperCycle CUDA Kernel Error] %s\n", cudaGetErrorString(kernelErr));
        return -1;
    }

    HC_CUDA_CHECK(cudaMemcpyAsync(g_ctx.h_pinned_pk, g_ctx.d_pk, total_bytes, cudaMemcpyDeviceToHost, g_ctx.stream));
    HC_CUDA_CHECK(cudaMemcpyAsync(g_ctx.h_pinned_sk, g_ctx.d_sk, total_bytes, cudaMemcpyDeviceToHost, g_ctx.stream));
    
    HC_CUDA_CHECK(cudaStreamSynchronize(g_ctx.stream));

    memcpy(pk_buffer, g_ctx.h_pinned_pk, total_bytes);
    memcpy(sk_buffer, g_ctx.h_pinned_sk, total_bytes);
    return 0;
}

static int cuda_generate_entropy_stub(uint64_t* seeds, uint8_t* out, size_t count) { 
    (void)seeds; (void)out; (void)count;
    return 0; 
}

static void cuda_teardown_impl(void) {
    if (g_ctx.stream) cudaStreamDestroy(g_ctx.stream);
    if (g_ctx.d_seeds) cudaFree(g_ctx.d_seeds);
    if (g_ctx.d_pk) cudaFree(g_ctx.d_pk);
    if (g_ctx.d_sk) cudaFree(g_ctx.d_sk);
    if (g_ctx.h_pinned_seeds) cudaFreeHost(g_ctx.h_pinned_seeds);
    // ... Reset global state ...
    cudaDeviceReset();
}

static hc_backend_t cuda_backend = {
    .backend_name = "NVIDIA CUDA (HyperCycle CQC - Level 5)",
    .generate_entropy = cuda_generate_entropy_stub,
    .cqc_keypair = cuda_cqc_keypair_impl,
    .teardown = cuda_teardown_impl
};

hc_backend_t* ns_cuda_backend_init(void) {
    int count;
    if (cudaGetDeviceCount(&count) != cudaSuccess || count == 0) return NULL;
    return &cuda_backend;
}
