#ifndef HC_HYPERCYCLE_H
#define HC_HYPERCYCLE_H

#include <stdint.h>
#include <stdbool.h>
#include <stddef.h>

/* --- Platform & Visibility Macros --- */

#if defined(_WIN32) || defined(__CYGWIN__)
    #ifdef HC_BUILD_LIB
        #define HC_API __declspec(dllexport)
    #else
        #define HC_API __declspec(dllimport)
    #endif
#elif defined(__GNUC__) && __GNUC__ >= 4
    #define HC_API __attribute__((visibility("default")))
#else
    #define HC_API
#endif

#ifdef __cplusplus
extern "C" {
#endif

/* --- Constants & Enums --- */

#define HC_API_VERSION 0x01060100 // v1.1.0

/* * Strict Error Codes
 * These allow the application to react intelligently to failure,
 * e.g., falling back to CPU if GPU memory is exhausted.
 */
typedef enum {
    HC_SUCCESS = 0,
    HC_ERR_NO_DEVICE = -1,
    HC_ERR_OUT_OF_MEMORY = -2,
    HC_ERR_KERNEL_FAILURE = -3,
    HC_ERR_INVALID_ARGS = -4,
    HC_ERR_VERSION_MISMATCH = -5,
    HC_ERR_NOT_INITIALIZED = -6,
    HC_ERR_OPT_NOT_SUPPORTED = -7,
    HC_ERR_DMA_FAILURE = -8
} hc_result_t;

/* * Hardware Types
 * Identifies the physical nature of the compute engine.
 */
typedef enum {
    HC_HW_CPU_ONLY = 0,
    HC_HW_NVIDIA_CUDA = 1,
    HC_HW_AMD_ROCM = 2
} hc_hardware_type_t;

/* * Optimization Engine Flags
 * Toggles for specific algorithmic "Boost" techniques.
 * Users may disable these for debugging or legacy compliance.
 */
typedef enum {
    HC_FLAG_NONE = 0,

    /* Functional Flags */
    HC_FLAG_ENABLE_BLINDING     = 1 << 0, // Enforce Vacuum Blinding (Side-channel protection)
    HC_FLAG_NON_BLOCKING        = 1 << 1, // Async execution (returns immediately)

    /* Algorithmic Optimizations */
    HC_FLAG_OPT_COMBINED_ROTATIONS = 1 << 2, // Pre-calc rotation products
    HC_FLAG_OPT_SANDWICH_ROTATION  = 1 << 3, // q*v*q^-1 (Gimbal safe)
    HC_FLAG_OPT_SCALAR_SPONGE      = 1 << 4, // Register-based chaos map
    HC_FLAG_OPT_BRANCHLESS_LOGIC   = 1 << 5, // Avoid 'if' in hot loops

    /* Hardware-Specific Optimizations */
    HC_FLAG_OPT_AVX512_IFMA        = 1 << 6, // Use Integer Fused Multiply-Add (Ice Lake+)

    /* Preset: Maximum Performance */
    HC_FLAG_OPT_MAX_PERFORMANCE    = (HC_FLAG_OPT_COMBINED_ROTATIONS | \
                                      HC_FLAG_OPT_SANDWICH_ROTATION | \
                                      HC_FLAG_OPT_SCALAR_SPONGE | \
                                      HC_FLAG_OPT_BRANCHLESS_LOGIC | \
                                      HC_FLAG_OPT_AVX512_IFMA)
} hc_gen_flags_t;

/* --- Configuration Structures --- */

/* * Context Configuration
 * Allows precise control over hardware resources and topology.
 * Includes padding to allow future expansion without breaking ABI.
 */
typedef struct {
    int device_id;           // GPU ID (e.g., 0, 1) or CPU Socket ID. -1 for Auto.
    int stream_priority;     // 0 = Default, 1 = High Priority (Latency critical)
    size_t memory_pool_size; // Pre-allocated device memory in bytes (0 = Auto)
    bool enable_profiling;   // Enable internal telemetry
    
    uint8_t _reserved[64];   // Padding for future parameters (e.g., affinity masks)
} hc_context_config_t;

/* * Capabilities Bitmask
 * Returned by hc_get_caps() to inform the user what the loaded backend supports.
 */
typedef struct {
    bool supports_dma;              // Can allocate pinned memory?
    bool supports_async;            // Can run non-blocking?
    bool supports_scalar_sponge;    // Has optimized scalar kernel?
    bool supports_avx512_ifma;      // Has AVX-512 Integer Fused Multiply-Add?
    uint32_t max_batch_size;        // Hardware limit
    uint32_t optimal_batch_size;    // Recommendation for peak throughput
    
    uint8_t _reserved[64];          // Padding for future capabilities
} hc_backend_caps_t;


/* --- Opaque Types --- */

/* * Backend Context Handle.
 * An opaque pointer allowing backends to store persistent state 
 * (e.g., cudaStream_t, allocated device memory) safely.
 */
typedef struct hc_context_s* hc_context_t;


/* --- The HyperCycle Backend Contract --- */

typedef struct {
    uint32_t api_version;       // ABI Version check (Must match HC_API_VERSION)
    const char* backend_name;   // e.g., "NVIDIA H100 (CUDA v12)"
    hc_hardware_type_t type;

    /* --- Lifecycle --- */

    /* Initialize with granular config */
    hc_result_t (*init_context)(hc_context_t* ctx, const hc_context_config_t* config);

    /* Release resources */
    void (*free_context)(hc_context_t ctx);

    /* Query Capabilities */
    hc_result_t (*get_caps)(hc_context_t ctx, hc_backend_caps_t* caps);

    /* --- Memory Management (Zero-Copy Optimization) --- */

    /* * Allocate Pinned Memory.
     * Returns host-accessible memory that is Page-Locked (Pinned) by the driver.
     * GPU can read/write this via DMA over PCIe, bypassing CPU staging buffers.
     */
    void* (*alloc_pinned)(hc_context_t ctx, size_t size);

    /* Free Pinned Memory */
    void (*free_pinned)(hc_context_t ctx, void* ptr);

    /* --- Execution --- */

    /* * Core Entropy Generation.
     * If out_buffer was allocated via alloc_pinned, transfer speed doubles.
     *
     * Params:
     * - ctx: Active context.
     * - seeds: Input entropy seeds (Host pointer).
     * - blinding_seeds: Optional masking seeds (Host pointer).
     * - out_buffer: Destination (Host/Pinned pointer).
     * - count: Number of keys.
     * - flags: optimization toggles.
     */
    hc_result_t (*generate_batch)(hc_context_t ctx, 
                                  const uint64_t* seeds, 
                                  const uint64_t* blinding_seeds, 
                                  uint8_t* out_buffer, 
                                  size_t count, 
                                  uint32_t flags);

    /* * Synchronize.
     * Block until all async operations in this context are complete.
     */
    hc_result_t (*sync)(hc_context_t ctx);

} hc_hypercycle_backend_t;


/* --- Public Manager API --- */

/* * Hardware Discovery
 * Scans the system for supported hardware libraries (libhc_cuda.so, etc.)
 * and loads the most capable one available.
 */
HC_API hc_hardware_type_t hc_hypercycle_detect_and_load(void);

/* * Initialization
 * Initializes the loaded backend with specific configuration.
 * Pass NULL for safe defaults.
 */
HC_API hc_result_t hc_hypercycle_init(const hc_context_config_t* config);

/* Feature Discovery */
HC_API hc_result_t hc_hypercycle_get_caps(hc_backend_caps_t* out_caps);

/* * Zero-Copy Allocator 
 * Use this wrapper to allocate 'out_buffer' for maximum DMA performance.
 */
HC_API void* hc_hypercycle_malloc(size_t size);
HC_API void hc_hypercycle_free(void* ptr);

/* * Execution Dispatcher
 * Routes the workload to the active high-performance backend.
 */
HC_API hc_result_t hc_hypercycle_dispatch(const uint64_t* seeds, 
                                          const uint64_t* blinding_seeds, 
                                          uint8_t* out_buffer, 
                                          size_t count, 
                                          hc_gen_flags_t flags);

/* Synchronization barrier */
HC_API hc_result_t hc_hypercycle_sync(void);

/* Shutdown */
HC_API void hc_hypercycle_shutdown(void);

#ifdef __cplusplus
}
#endif

#endif // HC_HYPERCYCLE_H
