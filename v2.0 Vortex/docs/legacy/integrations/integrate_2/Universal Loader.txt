#include "hc_gpu_universal.h"
#include "hc_math_core.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <dlfcn.h>

/* --- CPU Fallback --- */

static hc_gpu_status_t hc_cpu_initialize(void) {
    return HC_GPU_SUCCESS;
}

static void hc_cpu_shutdown(void) {
}

static hc_gpu_status_t hc_cpu_generate_entropy_batch(uint64_t seed_base, uint64_t blinding_seed, uint8_t* out_buffer, size_t count) {
    // OpenMP recommended here
    for (size_t i = 0; i < count; i++) {
        uint8_t local_key[32];
        hc_generate_single_key(seed_base, blinding_seed, i, local_key);
        memcpy(out_buffer + (i * 32), local_key, 32);
    }
    return HC_GPU_SUCCESS;
}

static const hc_gpu_backend_t g_cpu_backend = {
    .name = "HyperCycle CPU Core (Optimised)",
    .type = HC_BACKEND_CPU,
    .initialize = hc_cpu_initialize,
    .shutdown = hc_cpu_shutdown,
    .generate_entropy_batch = hc_cpu_generate_entropy_batch
};

/* --- Dynamic Loader --- */

typedef hc_gpu_status_t (*pf_init)(void);
typedef void (*pf_shut)(void);
typedef hc_gpu_status_t (*pf_gen)(uint64_t, uint64_t, uint8_t*, size_t);

static void* g_backend_handle = NULL;
static hc_gpu_backend_t g_active_backend;
static bool g_is_initialized = false;

static bool try_load_backend(const char* lib_name, 
                             const char* init_sym, 
                             const char* shut_sym, 
                             const char* gen_sym, 
                             hc_backend_type_t type, 
                             const char* display_name) {
    
    dlerror();
    void* handle = dlopen(lib_name, RTLD_LAZY | RTLD_LOCAL);
    if (!handle) return false;

    pf_init func_init = (pf_init)dlsym(handle, init_sym);
    pf_shut func_shut = (pf_shut)dlsym(handle, shut_sym);
    pf_gen  func_gen  = (pf_gen)dlsym(handle, gen_sym);

    if (!func_init || !func_shut || !func_gen) {
        dlclose(handle);
        return false;
    }

    if (func_init() != HC_GPU_SUCCESS) {
        dlclose(handle);
        return false;
    }

    g_backend_handle = handle;
    g_active_backend.name = display_name;
    g_active_backend.type = type;
    g_active_backend.initialize = func_init; 
    g_active_backend.shutdown = func_shut;
    g_active_backend.generate_entropy_batch = func_gen;
    
    return true;
}

const hc_gpu_backend_t* hc_gpu_auto_init(void) {
    if (g_is_initialized) return &g_active_backend;

    if (try_load_backend("./libhc_cuda.so", 
                         "hc_cuda_initialize", 
                         "hc_cuda_shutdown", 
                         "hc_cuda_generate_entropy_batch", 
                         HC_BACKEND_CUDA, 
                         "NVIDIA CUDA (HyperCycle)")) {
        g_is_initialized = true;
        return &g_active_backend;
    }

    if (try_load_backend("./libhc_rocm.so", 
                         "hc_rocm_initialize", 
                         "hc_rocm_shutdown", 
                         "hc_rocm_generate_entropy_batch", 
                         HC_BACKEND_ROCM, 
                         "AMD ROCm (HyperCycle)")) {
        g_is_initialized = true;
        return &g_active_backend;
    }

    g_active_backend = g_cpu_backend;
    g_is_initialized = true;
    return &g_active_backend;
}

void hc_gpu_shutdown(void) {
    if (g_is_initialized) {
        if (g_active_backend.shutdown) g_active_backend.shutdown();
        if (g_backend_handle) {
            dlclose(g_backend_handle);
            g_backend_handle = NULL;
        }
        g_is_initialized = false;
    }
}
