/*
 * HyperCycle v1.1 Origin - Blinding Test
 * ======================================
 */

#include <stdio.h>
#include <stdint.h>
#include <stdlib.h>
#include <string.h>
#include "hc_gpu_universal.h"

int main() {
    printf("--- HyperCycle v1.2 Blinding Test ---\n");

    const hc_gpu_backend_t* backend = hc_gpu_auto_init();
    if (!backend) return 1;

    printf("Backend: %s\n", backend->name);
    
    size_t count = 1;
    uint8_t buffer_raw[32];
    uint8_t buffer_blinded[32];
    
    uint64_t seed = 0xCAFEBABE;
    uint64_t mask = 0xDEADBEEF;

    // 1. Generate Raw Key (No Blinding)
    backend->generate_entropy_batch(seed, 0, buffer_raw, count);
    
    // 2. Generate Blinded Key (With Mask)
    backend->generate_entropy_batch(seed, mask, buffer_blinded, count);

    // 3. Compare
    if (memcmp(buffer_raw, buffer_blinded, 32) == 0) {
        printf("[FAIL] Blinded key matches raw key. Blinding inactive.\n");
    } else {
        printf("[PASS] Blinded key differs from raw key.\n");
    }

    hc_gpu_shutdown();
    return 0;
}
