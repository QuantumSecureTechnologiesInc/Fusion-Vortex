/* --- 1. Quaternion Arithmetic Implementation --- */

/*
 * Add two quaternions: R = A + B
 * Optimized for SIMD vectorization where available.
 */
HC_DEVICE_FUNC void hc_quat_add(const hc_quat_t *a, const hc_quat_t *b, hc_quat_t *r) {
    r->w = a->w + b->w;
    r->x = a->x + b->x;
    r->y = a->y + b->y;
    r->z = a->z + b->z;
}

/*
 * Subtract quaternions: R = A - B
 */
HC_DEVICE_FUNC void hc_quat_sub(const hc_quat_t *a, const hc_quat_t *b, hc_quat_t *r) {
    r->w = a->w - b->w;
    r->x = a->x - b->x;
    r->y = a->y - b->y;
    r->z = a->z - b->z;
}

/*
 * Hamilton Product: R = A * B
 * This is the engine of "Combined Rotations".
 *
 * Complexity: 16 multiplications, 12 additions.
 * Optimization: Uses fixed-point arithmetic to avoid FPU latency on some embedded targets.
 */
HC_DEVICE_FUNC void hc_quat_mul(const hc_quat_t *a, const hc_quat_t *b, hc_quat_t *r) {
    // Intermediate 64-bit math.
    // We maintain relative scale by dividing by HC_SCALE only once at the end.
    int64_t w = a->w * b->w - a->x * b->x - a->y * b->y - a->z * b->z;
    int64_t x = a->w * b->x + a->x * b->w + a->y * b->z - a->z * b->y;
    int64_t y = a->w * b->y - a->x * b->z + a->y * b->w + a->z * b->x;
    int64_t z = a->w * b->z + a->x * b->y - a->y * b->x + a->z * b->w;

    r->w = w / HC_SCALE;
    r->x = x / HC_SCALE;
    r->y = y / HC_SCALE;
    r->z = z / HC_SCALE;
}

/*
 * Quaternion Conjugate: R = A*
 * Essential for the sandwich product (rotation).
 */
HC_DEVICE_FUNC void hc_quat_conjugate(const hc_quat_t *a, hc_quat_t *r) {
    r->w =  a->w;
    r->x = -a->x;
    r->y = -a->y;
    r->z = -a->z;
}

/* --- 2. Rotation Optimizations --- */

/*
 * "Combine rotations": 
 * Allows pre-calculating a single quaternion representing multiple sequential rotations.
 * * Efficiency: 
 * - Sequential: v' = q2 * (q1 * v * q1*) * q2* (Requires 4 Hamilton products per vector)
 * - Combined:   q_net = q2 * q1; v' = q_net * v * q_net* (Requires 1 Mul once, then 2 per vector)
 * * Result: R_combined = B * A (Apply A first, then B)
 */
HC_DEVICE_FUNC void hc_quat_compose_rotations(const hc_quat_t *second_rot, const hc_quat_t *first_rot, hc_quat_t *combined_out) {
    hc_quat_mul(second_rot, first_rot, combined_out);
}

/*
 * 3D Rotation (Sandwich Product): R = q * v * q^-1
 *
 * "Avoid gimbal lock": 
 * Unlike Euler angles which can lose a degree of freedom at 90-degree pitch,
 * this quaternion operation maintains smooth interpolation on the 4D hypersphere surface.
 */
HC_DEVICE_FUNC void hc_quat_rotate(const hc_quat_t *rotator, const hc_quat_t *vector, hc_quat_t *result) {
    hc_quat_t conj;
    hc_quat_conjugate(rotator, &conj);
    
    hc_quat_t temp;
    hc_quat_mul(rotator, vector, &temp);
    hc_quat_mul(&temp, &conj, result);
}

/* --- 3. The "Sponge" Layer (Hash Accelerator) --- */

/*
 * Heisenberg-Euler Sponge Construction
 * State Size: 512 bits (2 Quaternions).
 * Capacity: 256 bits. Rate: 256 bits.
 */
typedef struct {
    hc_quat_t q_rate;     // Outer part (Rate)
    hc_quat_t q_capacity; // Inner part (Capacity)
} hc_sponge_t;

HC_DEVICE_FUNC void hc_sponge_init(hc_sponge_t *s) {
    s->q_rate.w = 0; s->q_rate.x = 0; s->q_rate.y = 0; s->q_rate.z = 0;
    // Seed capacity with physical constants to avoid zero-state attractors
    s->q_capacity.w = HC_ALPHA; 
    s->q_capacity.x = 0; s->q_capacity.y = 0; s->q_capacity.z = 0;
}

HC_DEVICE_FUNC void hc_sponge_permute(hc_sponge_t *s) {
    // 1. Cross-Couple
    s->q_rate.w ^= s->q_capacity.z;
    s->q_capacity.x ^= s->q_rate.y;

    // 2. Non-Linear Chaos Evolution (The Hash Accelerator Core)
    // Uses the 47-cycle Heisenberg-Euler map defined in hc_math_core.h
    for(int i=0; i<HC_CYCLES; i++) {
        hc_chaos_map_step(&s->q_rate);
        hc_chaos_map_step(&s->q_capacity);
        
        // Inter-cycle binding
        if (i % 7 == 0) {
            hc_quat_add(&s->q_rate, &s->q_capacity, &s->q_rate);
        }
    }
}

HC_DEVICE_FUNC void hc_sponge_absorb(hc_sponge_t *s, const uint8_t *data, size_t len) {
    size_t offset = 0;
    while(offset < len) {
        uint8_t *rate_bytes = (uint8_t*)&s->q_rate;
        size_t block_size = 32; // sizeof(hc_quat_t)
        
        for(size_t i=0; i < block_size && (offset + i) < len; i++) {
            rate_bytes[i] ^= data[offset + i];
        }
        
        hc_sponge_permute(s);
        offset += block_size;
    }
}

HC_DEVICE_FUNC void hc_sponge_squeeze(hc_sponge_t *s, uint8_t *out, size_t len) {
    size_t offset = 0;
    while(offset < len) {
        hc_sponge_permute(s);
        
        uint8_t *rate_bytes = (uint8_t*)&s->q_rate;
        size_t block_size = 32;
        
        for(size_t i=0; i < block_size && (offset + i) < len; i++) {
            out[offset + i] = rate_bytes[i];
        }
        offset += block_size;
    }
}
