/*
 * HyperCycle v1.1 Origin - Interface Contract
 * ===========================================
 *
 * This header defines the high-performance plugin architecture for HyperCycle.
 * It acts as the bridge between the application layer and the hardware-specific
 * backends (CUDA/ROCm/CPU), enforcing the "Optimization Document" standards.
 *
 * Key Enhancements (v1.1):
 * 1. Persistent Contexts: Allows zero-allocation loops by caching GPU buffers.
 * 2. Blinding Support: Explicit parameters for side-channel masking seeds.
 * 3. Performance Profiles: Hinting system for Latency vs. Throughput tuning.
 *
 * Author: Senior Cryptographic Engineer
 * Version: 1.1 (Optimised Interface)
 */

#ifndef HC_HYPERCYCLE_H
#define HC_HYPERCYCLE_H

#include <stdint.h>
#include <stdbool.h>
#include <stddef.h>

#ifdef __cplusplus
extern "C" {
#endif

/* --- Constants & Enums --- */

#define HC_API_VERSION 0x01040000 // v1.1.0

/* Strict Error Codes */
typedef enum {
    HC_SUCCESS = 0,
    HC_ERR_NO_DEVICE = -1,
    HC_ERR_OUT_OF_MEMORY = -2,
    HC_ERR_KERNEL_FAILURE = -3,
    HC_ERR_INVALID_ARGS = -4,
    HC_ERR_VERSION_MISMATCH = -5,
    HC_ERR_NOT_INITIALIZED = -6
} hc_result_t;

/* Hardware Types */
typedef enum {
    HC_HW_CPU_ONLY = 0,
    HC_HW_NVIDIA_CUDA = 1,
    HC_HW_AMD_ROCM = 2
} hc_hardware_type_t;

/* Performance Hints for Context Initialization */
typedef enum {
    HC_PERF_DEFAULT = 0,
    HC_PERF_LOW_LATENCY = 1,   // Optimise for HFT (e.g., Spinlocks, persistent kernels)
    HC_PERF_MAX_THROUGHPUT = 2 // Optimise for Batch/IoT (e.g., large buffers)
} hc_perf_profile_t;

/* Generation Flags */
typedef enum {
    HC_FLAG_NONE = 0,
    HC_FLAG_ENABLE_BLINDING = 1 << 0, // Enforce Vacuum Blinding (Side-channel protection)
    HC_FLAG_NON_BLOCKING    = 1 << 1  // Async execution (if supported)
} hc_gen_flags_t;

/* --- Opaque Types --- */

/* * Backend Context Handle.
 * Allows backends to store persistent state (e.g., cudaStream_t, allocated device memory)
 * to avoid setup/teardown overhead on every call.
 */
typedef struct hc_context_s* hc_context_t;


/* --- The HyperCycle Backend Contract --- */

typedef struct {
    uint32_t api_version;       // ABI Version check
    const char* backend_name;   // e.g., "NVIDIA H100 (CUDA v12)"
    hc_hardware_type_t type;

    /* * Initialize Context
     * Allocates resources based on the requested performance profile.
     */
    hc_result_t (*init_context)(hc_context_t* ctx, hc_perf_profile_t profile);

    /* * Free Context
     * Releases persistent resources.
     */
    void (*free_context)(hc_context_t ctx);

    /* * Core Entropy Generation (Optimised Batch)
     * * ctx:            Active context handle.
     * seeds:          Array of 64-bit entropy seeds (one per item).
     * blinding_seeds: Array of 64-bit masking seeds (optional, can be NULL if flags allow).
     * out_buffer:     Destination for entropy (32 bytes per item).
     * count:          Number of items to generate.
     * flags:          Operational toggles (Blinding, Async).
     */
    hc_result_t (*generate_batch)(hc_context_t ctx, 
                                  const uint64_t* seeds, 
                                  const uint64_t* blinding_seeds, 
                                  uint8_t* out_buffer, 
                                  size_t count, 
                                  uint32_t flags);

} hc_hypercycle_backend_t;


/* --- Public Manager API --- */

/*
 * Detects available hardware and loads the optimal backend plugin.
 * Returns the hardware type engaged.
 */
hc_hardware_type_t hc_hypercycle_detect_and_load(void);

/*
 * Initializes the global active backend context.
 * Must be called after detect_and_load.
 */
hc_result_t hc_hypercycle_init(hc_perf_profile_t profile);

/*
 * High-Level Dispatcher
 * Routes the request to the active backend context.
 */
hc_result_t hc_hypercycle_dispatch(const uint64_t* seeds, 
                                   const uint64_t* blinding_seeds, 
                                   uint8_t* out_buffer, 
                                   size_t count, 
                                   hc_gen_flags_t flags);

/*
 * Global Shutdown
 * Cleans up context and unloads plugins.
 */
void hc_hypercycle_shutdown(void);

#ifdef __cplusplus
}
#endif

#endif // HC_HYPERCYCLE_H
