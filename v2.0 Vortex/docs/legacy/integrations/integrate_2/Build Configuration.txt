cmake_minimum_required(VERSION 3.21)

# Project: HyperCycle
# Version: 1.0.0 (Genesis)
# Languages: C (Loader), CXX (HIP/ROCm)
# Note: CUDA is enabled optionally via check_language to allow graceful degradation.
project(HyperCycle 
    VERSION 1.0.0
    DESCRIPTION "HyperCycle Genesis: Universal GPU Hardware Abstraction Layer"
    LANGUAGES C CXX
)

# --- Global Configuration ---

# 1. Path Standards
# Use GNUInstallDirs to ensure we adhere to standard Linux hierarchy (lib/ vs lib64/)
include(GNUInstallDirs)

# 2. Build Options
option(HC_ENABLE_CUDA "Enable compilation of the NVIDIA CUDA plugin" ON)
option(HC_ENABLE_ROCM "Enable compilation of the AMD ROCm/HIP plugin" ON)
option(HC_BUILD_TESTS "Build internal test suite" OFF)

# 3. Security & Optimisation
# Mandatory Position Independent Code for dynamic loading context
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

# Default to Release if not specified to ensure -O3 is applied in CI/CD
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

# Output directories: Plugins live next to the core lib in the build tree
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

message(STATUS "HyperCycle Genesis v${PROJECT_VERSION}: Configuring for ${CMAKE_SYSTEM_NAME} (${CMAKE_BUILD_TYPE})")

# --- 1. HyperCycle Core (The Loader & CPU Engine) ---

# The core library handles runtime detection (dlopen) and AVX-512 fallbacks.
add_library(hypercycle_core SHARED
    hc_core.c
    # In a full source tree, hc_cpu_avx.c would be included here
)

# Create a namespaced alias for clean consumption by downstream CMake projects
add_library(HyperCycle::Core ALIAS hypercycle_core)

# Production Versioning (libhypercycle_core.so.1.0.0)
set_target_properties(hypercycle_core PROPERTIES
    VERSION ${PROJECT_VERSION}
    SOVERSION ${PROJECT_VERSION_MAJOR}
    PUBLIC_HEADER "hc_core.h"
)

# Core Compiler Definitions
# We pass both the INSTALL path and the BUILD path.
# This ensures the loader can find plugins "straight away" in the build tree
# without needing to run 'make install' during development.
target_compile_definitions(hypercycle_core PRIVATE 
    HC_PLUGIN_PATH="${CMAKE_INSTALL_FULL_LIBDIR}/hypercycle"
    HC_BUILD_PLUGIN_PATH="${CMAKE_BINARY_DIR}/lib/hypercycle"
    HC_VERSION_MAJOR=${PROJECT_VERSION_MAJOR}
    HC_VERSION_MINOR=${PROJECT_VERSION_MINOR}
    _GNU_SOURCE # Required for dladdr and other POSIX extensions
)

# Hardened Compiler Flags (GCC/Clang)
if(CMAKE_C_COMPILER_ID MATCHES "GNU|Clang")
    target_compile_options(hypercycle_core PRIVATE 
        -O3 
        -march=native 
        -Wall 
        -Wextra 
        -Wpedantic 
        -Wconversion 
        -Wshadow
        -Wformat=2
        -fvisibility=hidden     # Hide internal symbols
        -fstack-protector-strong # Buffer overflow protection
    )
    # Fortify Source in Release mode
    if(CMAKE_BUILD_TYPE STREQUAL "Release")
        target_compile_definitions(hypercycle_core PRIVATE _FORTIFY_SOURCE=2)
    endif()
endif()

# Dependencies
target_link_libraries(hypercycle_core PRIVATE ${CMAKE_DL_LIBS})

# Include Directories (Build interface vs Install interface)
target_include_directories(hypercycle_core PUBLIC 
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
    $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>
)

# --- 2. NVIDIA CUDA Plugin (Optional) ---

if(HC_ENABLE_CUDA)
    include(CheckLanguage)
    check_language(CUDA)

    if(CMAKE_CUDA_COMPILER)
        enable_language(CUDA)
        message(STATUS "  [+] CUDA detected. Configuring NVIDIA Plugin (hc_cuda).")

        # MODULE library: Meant for dlopen(), not linking.
        add_library(hc_cuda MODULE hc_vacuum_gpu.cu)

        # Allow packagers to override architectures via -DHC_CUDA_ARCHS="80;90"
        set(HC_CUDA_ARCHS "80;86;89;90" CACHE STRING "Target CUDA Architectures (Ampere+)")

        set_target_properties(hc_cuda PROPERTIES 
            OUTPUT_NAME "hc_cuda"
            PREFIX "lib"
            SUFFIX ".so"
            CUDA_ARCHITECTURES "${HC_CUDA_ARCHS}"
            CUDA_SEPARABLE_COMPILATION ON
            LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib/hypercycle" # Organise in subdir
        )

        target_link_libraries(hc_cuda PRIVATE CUDA::cudart)
        
        # Install component tagging
        install(TARGETS hc_cuda 
            LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}/hypercycle
            COMPONENT plugins
        )
    else()
        message(STATUS "  [-] CUDA compiler not found. Skipping NVIDIA Plugin.")
    endif()
endif()

# --- 3. AMD ROCm/HIP Plugin (Optional) ---

if(HC_ENABLE_ROCM)
    # Attempt to find HIP (ROCm)
    find_package(HIP QUIET)

    if(HIP_FOUND)
        message(STATUS "  [+] HIP/ROCm detected. Configuring AMD Plugin (hc_rocm).")
        
        # HIP typically uses the CXX compiler (hipcc)
        add_library(hc_rocm MODULE hc_vacuum_amd.hip.cpp)

        set_target_properties(hc_rocm PROPERTIES 
            OUTPUT_NAME "hc_rocm"
            PREFIX "lib"
            SUFFIX ".so"
            LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib/hypercycle"
        )

        target_link_libraries(hc_rocm PRIVATE HIP::hip_runtime)
        
        # Optimisation flags for AMD
        target_compile_options(hc_rocm PRIVATE -O3 -munsafe-fp-atomics)

        install(TARGETS hc_rocm 
            LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}/hypercycle
            COMPONENT plugins
        )
    else()
        message(STATUS "  [-] HIP/ROCm not found. Skipping AMD Plugin.")
    endif()
endif()

# --- 4. Installation & Exporting ---

# A. Install the Core Library
install(TARGETS hypercycle_core
    EXPORT HyperCycleTargets
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    PUBLIC_HEADER DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
    COMPONENT core
)

# B. Export the Targets (allows 'find_package(HyperCycle)' in other projects)
install(EXPORT HyperCycleTargets
    FILE HyperCycleTargets.cmake
    NAMESPACE HyperCycle::
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/HyperCycle
)

# C. Generate Config Version File
include(CMakePackageConfigHelpers)
write_basic_package_version_file(
    "HyperCycleConfigVersion.cmake"
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY SameMajorVersion
)

# D. Create the Config File
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/HyperCycleConfig.cmake"
"include(\"\${CMAKE_CURRENT_LIST_DIR}/HyperCycleTargets.cmake\")
")

# E. Install Config Files
install(FILES
    "${CMAKE_CURRENT_BINARY_DIR}/HyperCycleConfig.cmake"
    "${CMAKE_CURRENT_BINARY_DIR}/HyperCycleConfigVersion.cmake"
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/HyperCycle
)

# --- Summary ---
message(STATUS "-----------------------------------------------------------")
message(STATUS "HyperCycle Genesis Configuration Complete")
message(STATUS "  Version:      ${PROJECT_VERSION}")
message(STATUS "  Install Path: ${CMAKE_INSTALL_PREFIX}")
message(STATUS "  Plugin Path:  ${CMAKE_INSTALL_FULL_LIBDIR}/hypercycle")
message(STATUS "-----------------------------------------------------------")
