#ifndef HC_GPU_UNIVERSAL_H
#define HC_GPU_UNIVERSAL_H

#include <stdint.h>
#include <stddef.h>
#include <stdbool.h>

/* * Status Codes 
 * Used to propagate errors from deep within GPU runtimes to the host application.
 */
typedef enum {
    HC_GPU_SUCCESS = 0,
    HC_GPU_ERR_NO_DEVICE = -1,
    HC_GPU_ERR_MEMORY = -2,
    HC_GPU_ERR_KERNEL_FAILURE = -3,
    HC_GPU_ERR_NOT_INITIALIZED = -4,
    HC_GPU_ERR_SYMBOL_MISSING = -5
} hc_gpu_status_t;

/* * Backend Types 
 * Identifies which hardware engine is currently active.
 */
typedef enum {
    HC_BACKEND_CPU = 0,
    HC_BACKEND_CUDA = 1,
    HC_BACKEND_ROCM = 2
} hc_backend_type_t;

/*
 * The Backend V-Table
 * This virtual table structure allows the main application to be 
 * completely agnostic of the underlying hardware.
 */
typedef struct {
    const char* name;
    hc_backend_type_t type;
    
    /* Function Pointers */
    
    /* Initialize the backend (e.g., check device count, set device). */
    hc_gpu_status_t (*initialize)(void);
    
    /* Clean up resources (e.g., reset device). */
    void (*shutdown)(void);
    
    /* * Core Entropy Generator (Blinded)
     * seed_base: The master seed (system entropy).
     * blinding_seed: A secondary seed used to generate a vacuum mask.
     * If 0, no masking is applied (raw output).
     * If non-zero, Output = Chaos(seed) ^ Chaos(blinding_seed).
     * out_buffer: Destination for keys (must be allocated by caller).
     * count: Number of keys to generate (each key is 32 bytes).
     */
    hc_gpu_status_t (*generate_entropy_batch)(uint64_t seed_base, uint64_t blinding_seed, uint8_t* out_buffer, size_t count);
    
} hc_gpu_backend_t;

#ifdef __cplusplus
extern "C" {
#endif

/* * Auto-Discovery (Implemented in hc_gpu_universal.c)
 * Scans for available drivers/libraries and returns the pointer to the 
 * active backend V-Table.
 */
const hc_gpu_backend_t* hc_gpu_auto_init(void);

/* * Global Shutdown
 * Cleans up the active backend and releases dynamic library handles.
 */
void hc_gpu_shutdown(void);

#ifdef __cplusplus
}
#endif

#endif /* HC_GPU_UNIVERSAL_H */
