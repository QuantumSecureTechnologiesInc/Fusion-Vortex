/*
 * HyperCycle v1.1 "Origin" - Core Hardware Abstraction Layer
 * * Module: Chaos-Quaternion Cryptography (CQC) Interface
 * * Standard: NIST Level 5 (AES-256 Equivalent)
 * * Status: Production Ready
 *
 * Design Rationale:
 * This header establishes the contract for "Pure Chaos" cryptography.
 * It enforces the 512-byte key topology required for quantum resistance
 * and handles symbol visibility for shared library distribution.
 */

#ifndef HC_CORE_H
#define HC_CORE_H

#include <stdint.h>
#include <stddef.h>
#include <stdbool.h>

// --- Symbol Visibility Macros ---
// Ensure we only export the public API, reducing binary size and conflicts.
#if defined(_WIN32) || defined(__CYGWIN__)
  #ifdef BUILDING_HC_CORE
    #define HC_API __declspec(dllexport)
  #else
    #define HC_API __declspec(dllimport)
  #endif
#else
  #define HC_API __attribute__((visibility("default")))
#endif

#ifdef __cplusplus
extern "C" {
#endif

// --- Versioning ---
#define HC_VERSION_MAJOR 1
#define HC_VERSION_MINOR 0
#define HC_VERSION_PATCH 0

// --- CQC NIST Level 5 Constants ---
// Security Claim: >256 bits of quantum security (Grover resistance)
#define HC_CQC_KEY_SIZE   512  // 4096 bits per key
#define HC_CQC_SEED_SIZE  512  // Symmetric entropy input

// Hardware Identification Enums
typedef enum {
    HC_HW_CPU_ONLY = 0,
    HC_HW_NVIDIA_CUDA = 1,
    HC_HW_AMD_ROCM = 2,
    HC_HW_ERROR = -1
} hc_hardware_type_t;

// The "HyperCycle Contract"
// Every backend plugin must implement and return this structure.
typedef struct {
    const char* backend_name;

    // Legacy Entropy Generator (Raw Randomness)
    int (*generate_entropy)(uint64_t* seeds, uint8_t* out_buffer, size_t count);

    // CQC Key Pair Generator (NIST Level 5)
    // Generates 'count' keypairs.
    // pk_buffer and sk_buffer must be size (count * HC_CQC_KEY_SIZE).
    int (*cqc_keypair)(uint64_t* seeds, uint8_t* pk_buffer, uint8_t* sk_buffer, size_t count);

    void (*teardown)(void);

} hc_backend_t;

/* * Public API - The HyperCycle Manager * */

// Detects available hardware and loads the best plugin (CUDA/ROCm).
HC_API hc_hardware_type_t hc_detect_and_load(void);

// Shuts down the active backend.
HC_API void hc_shutdown(void);

// Returns the name of the currently active backend.
HC_API const char* hc_get_backend_name(void);

// --------------------------------------------------------------------------
// Primary CQC API
// --------------------------------------------------------------------------

/**
 * @brief Generates NIST Level 5 Chaos-Quaternion Keypairs.
 * * Uses the Heisenberg-Euler chaos map to evolve private entropy (seeds)
 * into a public chaotic state.
 *
 * @param seeds      Input entropy (flat array of uint64_t).
 * Size: count * (HC_CQC_KEY_SIZE / 8) items.
 * @param pk_buffer  Output Public Keys. Size: count * 512 bytes.
 * @param sk_buffer  Output Secret Keys. Size: count * 512 bytes.
 * @param count      Number of keypairs to generate.
 * @return 0 on success, -1 on backend error, -2 on invalid input.
 */
HC_API int hc_cqc_keypair(uint64_t* seeds, uint8_t* pk_buffer, uint8_t* sk_buffer, size_t count);

#ifdef __cplusplus
}
#endif

#endif // HC_CORE_H
